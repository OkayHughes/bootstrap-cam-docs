23 subroutine MPP_DO_UPDATE_AD_3D_( f_addrs, domain, update, d_type, ke, flags)
24!updates data domain of 3D field whose computational domains have been computed
25 integer(LONG_KIND), intent(in) :: f_addrs(:,:)
26 type(domain2D), intent(in) :: domain
27 type(overlapSpec), intent(in) :: update
28 MPP_TYPE_, intent(in) :: d_type ! creates unique interface
29 integer, intent(in) :: ke
30 integer, optional, intent(in) :: flags
31
32 MPP_TYPE_ :: field(update%xbegin:update%xend, update%ybegin:update%yend,ke)
33 pointer(ptr_field, field)
34 integer :: update_flags
35 type(overlap_type), pointer :: overPtr => NULL()
36 character(len=8) :: text
37
38!equate to mpp_domains_stack
39 MPP_TYPE_ :: buffer(size(mpp_domains_stack(:)))
40 pointer( ptr, buffer )
41 integer :: buffer_pos
42
43!receive domains saved here for unpacking
44!for non-blocking version, could be recomputed
45 integer, allocatable :: msg1(:), msg2(:)
46 logical :: send(8), recv(8), update_edge_only
47 integer :: to_pe, from_pe, pos, msgsize, msgsize_send
48 integer :: n, l_size, l, m, i, j, k
49 integer :: is, ie, js, je, tMe, dir
50 integer :: start, start1, start2, index, is1, ie1, js1, je1, ni, nj, total
51 integer :: buffer_recv_size, nlist, outunit
52 integer :: send_start_pos
53 integer :: send_msgsize(MAXLIST)
54
55
56 outunit = stdout()
57 ptr = LOC(mpp_domains_stack)
58 l_size = size(f_addrs,1)
59
60 update_flags = XUPDATE+YUPDATE !default
61 if( PRESENT(flags) )update_flags = flags
62
63 update_edge_only = BTEST(update_flags, EDGEONLY)
64 recv(1) = BTEST(update_flags,EAST)
65 recv(3) = BTEST(update_flags,SOUTH)
66 recv(5) = BTEST(update_flags,WEST)
67 recv(7) = BTEST(update_flags,NORTH)
68 if( update_edge_only ) then
69 if( .NOT. (recv(1) .OR. recv(3) .OR. recv(5) .OR. recv(7)) ) then
70 recv(1) = .true.
71 recv(3) = .true.
72 recv(5) = .true.
73 recv(7) = .true.
74 endif
75 else
76 recv(2) = recv(1) .AND. recv(3)
77 recv(4) = recv(3) .AND. recv(5)
78 recv(6) = recv(5) .AND. recv(7)
79 recv(8) = recv(7) .AND. recv(1)
80 endif
81 send = recv
82
83 if(debug_message_passing) then
84 nlist = size(domain%list(:))
85 allocate(msg1(0:nlist-1), msg2(0:nlist-1) )
86 msg1 = 0
87 msg2 = 0
88 do m = 1, update%nrecv
89 overPtr => update%recv(m)
90 msgsize = 0
91 do n = 1, overPtr%count
92 dir = overPtr%dir(n)
93 if(recv(dir)) then
94 is = overPtr%is(n); ie = overPtr%ie(n)
95 js = overPtr%js(n); je = overPtr%je(n)
96 msgsize = msgsize + (ie-is+1)*(je-js+1)
97 end if
98 end do
99 from_pe = update%recv(m)%pe
100 l = from_pe-mpp_root_pe()
101 call mpp_recv( msg1(l), glen=1, from_pe=from_pe, block=.FALSE., tag=COMM_TAG_1 )
102 msg2(l) = msgsize
103 enddo
104
105 do m = 1, update%nsend
106 overPtr => update%send(m)
107 msgsize = 0
108 do n = 1, overPtr%count
109 dir = overPtr%dir(n)
110 if(send(dir)) then
111 is = overPtr%is(n); ie = overPtr%ie(n)
112 js = overPtr%js(n); je = overPtr%je(n)
113 msgsize = msgsize + (ie-is+1)*(je-js+1)
114 end if
115 end do
116 call mpp_send( msgsize, plen=1, to_pe=overPtr%pe, tag=COMM_TAG_1 )
117 enddo
118 call mpp_sync_self(check=EVENT_RECV)
119
120 do m = 0, nlist-1
121 if(msg1(m) .NE. msg2(m)) then
122 print*, "My pe = ", mpp_pe(), ",domain name =", trim(domain%name), ",from pe=", &
123 domain%list(m)%pe, ":send size = ", msg1(m), ", recv size = ", msg2(m)
124 call mpp_error(FATAL, "mpp_do_update: mismatch on send and recv size")
125 endif
126 enddo
127 call mpp_sync_self()
128 write(outunit,*)"NOTE from mpp_do_update: message sizes are matched between send and recv for domain " &
129 //trim(domain%name)
130 deallocate(msg1, msg2)
131 endif
132
133 !recv
134 buffer_pos = 0
135 do m = 1, update%nrecv
136 overPtr => update%recv(m)
137 if( overPtr%count == 0 )cycle
138 msgsize = 0
139 do n = 1, overPtr%count
140 dir = overPtr%dir(n)
141 if(recv(dir)) then
142 tMe = overPtr%tileMe(n)
143 is = overPtr%is(n); ie = overPtr%ie(n)
144 js = overPtr%js(n); je = overPtr%je(n)
145 msgsize = msgsize + (ie-is+1)*(je-js+1)
146 msgsize_send = (ie-is+1)*(je-js+1)*ke*l_size
147 pos = buffer_pos + msgsize_send
148 do l=1,l_size ! loop over number of fields
149 ptr_field = f_addrs(l, tMe)
150 do k = ke,1,-1
151 do j = je, js, -1
152 do i = ie, is, -1
153 buffer(pos) = field(i,j,k)
154 field(i,j,k) = 0.
155 pos = pos - 1
156 end do
157 end do
158 end do
159 end do
160 end if
161 end do
162
163 msgsize = msgsize*ke*l_size
164 if( msgsize.GT.0 )then
165 to_pe = overPtr%pe
166 call mpp_send( buffer(buffer_pos+1), plen=msgsize, to_pe=to_pe, tag=COMM_TAG_2 )
167 buffer_pos = buffer_pos + msgsize
168 end if
169 end do ! end do m = 1, update%nrecv
170 buffer_recv_size = buffer_pos
171
172 ! send
173 do m = 1, update%nsend
174 overPtr => update%send(m)
175 if( overPtr%count == 0 )cycle
176 pos = buffer_pos
177 msgsize = 0
178 do n = 1, overPtr%count
179 dir = overPtr%dir(n)
180 if( send(dir) ) msgsize = msgsize + overPtr%msgsize(n)
181 enddo
182 if( msgsize.GT.0 )then
183 msgsize = msgsize*ke*l_size
184 msgsize_send = msgsize
185 from_pe = overPtr%pe
186 call mpp_recv( buffer(buffer_pos+1), glen=msgsize, from_pe=from_pe, block=.FALSE., tag=COMM_TAG_2 )
187 buffer_pos = buffer_pos + msgsize
188 end if
189 end do ! end do ist = 0,nlist-1
190
191 call mpp_sync_self(check=EVENT_RECV)
192
193 buffer_pos = buffer_recv_size
194
195 ! send
196 do m = 1, update%nsend
197 overPtr => update%send(m)
198 if( overPtr%count == 0 )cycle
199 pos = buffer_pos
200 msgsize = 0
201 do n = 1, overPtr%count
202 dir = overPtr%dir(n)
203 if( send(dir) ) msgsize = msgsize + overPtr%msgsize(n)
204 enddo
205 if( msgsize.GT.0 )then
206 buffer_pos = pos
207 end if
208
209 do n = 1, overPtr%count
210 dir = overPtr%dir(n)
211 if( send(dir) ) then
212 tMe = overPtr%tileMe(n)
213 is = overPtr%is(n); ie = overPtr%ie(n)
214 js = overPtr%js(n); je = overPtr%je(n)
215 select case( overPtr%rotation(n) )
216 case(ZERO)
217 do l=1,l_size ! loop over number of fields
218 ptr_field = f_addrs(l, tMe)
219 do k = 1,ke
220 do j = js, je
221 do i = is, ie
222 pos = pos + 1
223 field(i,j,k)=field(i,j,k)+buffer(pos)
224 end do
225 end do
226 end do
227 end do
228 case( MINUS_NINETY )
229 do l=1,l_size ! loop over number of fields
230 ptr_field = f_addrs(l, tMe)
231 do k = 1,ke
232 do i = is, ie
233 do j = je, js, -1
234 pos = pos + 1
235 field(i,j,k)=field(i,j,k)+buffer(pos)
236 end do
237 end do
238 end do
239 end do
240 case( NINETY )
241 do l=1,l_size ! loop over number of fields
242 ptr_field = f_addrs(l, tMe)
243 do k = 1,ke
244 do i = ie, is, -1
245 do j = js, je
246 pos = pos + 1
247 field(i,j,k)=field(i,j,k)+buffer(pos)
248 end do
249 end do
250 end do
251 end do
252 case( ONE_HUNDRED_EIGHTY )
253 do l=1,l_size ! loop over number of fields
254 ptr_field = f_addrs(l, tMe)
255 do k = 1,ke
256 do j = je, js, -1
257 do i = ie, is, -1
258 pos = pos + 1
259 field(i,j,k)=field(i,j,k)+buffer(pos)
260 end do
261 end do
262 end do
263 end do
264 end select
265 endif
266 end do ! do n = 1, overPtr%count
267
268 msgsize = pos - buffer_pos
269 if( msgsize.GT.0 )then
270 buffer_pos = pos
271 end if
272 end do ! end do ist = 0,nlist-1
273
274 call mpp_sync_self()
275
276 return
277 end subroutine MPP_DO_UPDATE_AD_3D_

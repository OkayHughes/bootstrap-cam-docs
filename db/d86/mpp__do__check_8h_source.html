20 subroutine MPP_DO_CHECK_3D_( f_addrs, domain, check, d_type, ke, flags, name)
21!updates data domain of 3D field whose computational domains have been computed
22 integer(LONG_KIND), intent(in) :: f_addrs(:,:)
23 type(domain2D), intent(in) :: domain
24 type(overlapSpec), intent(in) :: check
25 MPP_TYPE_, intent(in) :: d_type ! creates unique interface
26 integer, intent(in) :: ke
27 integer, optional, intent(in) :: flags
28 character(len=*), optional, intent(in) :: name
29
30 MPP_TYPE_ :: field(check%xbegin:check%xend, check%ybegin:check%yend,ke)
31 pointer(ptr_field, field)
32 integer :: update_flags
33 character(len=8) :: text
34 character(len=64) :: field_name
35
36!equate to mpp_domains_stack
37 MPP_TYPE_ :: buffer(size(mpp_domains_stack(:)))
38 pointer( ptr, buffer )
39 integer :: buffer_pos
40 integer, allocatable :: msg1(:), msg2(:)
41!receive domains saved here for unpacking
42!for non-blocking version, could be recomputed
43 integer :: to_pe, from_pe, pos, msgsize
44 integer :: n, l_size, l, m, i, j, k
45 integer :: is, ie, js, je, tMe
46 integer :: buffer_recv_size, nlist
47 integer :: outunit
48
49 outunit = stdout()
50 ptr = LOC(mpp_domains_stack)
51 l_size = size(f_addrs,1)
52
53 update_flags = XUPDATE+YUPDATE !default
54 if( PRESENT(flags) )update_flags = flags
55
56 !--- if debug_update_level is not NO_DEBUG, check the consistency on the bounds
57 !--- (domain is symmetry or folded north edge). North bound will be checked when north edge is folded.
58 !--- when domain is symmetry, For data on T-cell, no check is needed; for data on E-cell,
59 !--- data on East and West boundary will be checked ; For data on N-cell, data on North and South
60 !--- boundary will be checked; For data on C-cell, data on West, East, South, North will be checked.
61 !--- The check will be done in the following way: Western boundary data sent to Eastern boundary to check
62 !--- and Southern boundary to check
63
64 if(present(name)) then
65 field_name = name
66 else
67 field_name = "un-named"
68 end if
69
70 if(debug_message_passing) then
71 nlist = size(domain%list(:))
72 allocate(msg1(0:nlist-1), msg2(0:nlist-1) )
73 msg1 = 0
74 msg2 = 0
75 do m = 1, check%nrecv
76 msgsize = 0
77 do n = 1, check%recv(m)%count
78 is = check%recv(m)%is(n); ie = check%recv(m)%ie(n)
79 js = check%recv(m)%js(n); je = check%recv(m)%je(n)
80 msgsize = msgsize + (ie-is+1)*(je-js+1)
81 end do
82 from_pe = check%recv(m)%pe
83 l = from_pe-mpp_root_pe()
84 call mpp_recv( msg1(l), glen=1, from_pe=from_pe, block=.FALSE., tag=COMM_TAG_1)
85 msg2(l) = msgsize
86 enddo
87
88 do m = 1, check%nsend
89 msgsize = 0
90 do n = 1, check%send(m)%count
91 is = check%send(m)%is(n); ie = check%send(m)%ie(n)
92 js = check%send(m)%js(n); je = check%send(m)%je(n)
93 msgsize = msgsize + (ie-is+1)*(je-js+1)
94 end do
95 call mpp_send(msgsize, plen=1, to_pe=check%send(m)%pe, tag=COMM_TAG_1)
96 enddo
97 call mpp_sync_self(check=EVENT_RECV)
98
99 do m = 0, nlist-1
100 if(msg1(m) .NE. msg2(m)) then
101 print*, "My pe = ", mpp_pe(), ",domain name =", trim(domain%name), ",from pe=", &
102 domain%list(m)%pe, ":send size = ", msg1(m), ", recv size = ", msg2(m)
103 call mpp_error(FATAL, "mpp_do_check: mismatch on send and recv size")
104 endif
105 enddo
106 call mpp_sync_self()
107 write(outunit,*)"NOTE from mpp_do_check: message sizes are matched between send and recv for domain " &
108 //trim(domain%name)
109 deallocate(msg1, msg2)
110 endif
111
112 buffer_pos = 0
113 !--- pre-post recv the data
114 do m = 1, check%nrecv
115 msgsize = 0
116 do n = 1, check%recv(m)%count
117 is = check%recv(m)%is(n); ie = check%recv(m)%ie(n)
118 js = check%recv(m)%js(n); je = check%recv(m)%je(n)
119 msgsize = msgsize + (ie-is+1)*(je-js+1)
120 end do
121 msgsize = msgsize*ke*l_size
122
123 if( msgsize.GT.0 )then
124 from_pe = check%recv(m)%pe
125 mpp_domains_stack_hwm = max( mpp_domains_stack_hwm, (buffer_pos+msgsize) )
126 if( mpp_domains_stack_hwm.GT.mpp_domains_stack_size )then
127 write( text,'(i8)' )mpp_domains_stack_hwm
128 call mpp_error( FATAL, 'MPP_DO_CHECK: mpp_domains_stack overflow, '// &
129 'call mpp_domains_set_stack_size('//trim(text)//') from all PEs.' )
130 end if
131 call mpp_recv( buffer(buffer_pos+1), glen=msgsize, from_pe=from_pe, block=.FALSE., tag=COMM_TAG_2 )
132 buffer_pos = buffer_pos + msgsize
133 end if
134 end do
135 buffer_recv_size = buffer_pos
136
137 !--- send the data
138 do m = 1, check%nsend
139 pos = buffer_pos
140 do n = 1, check%recv(m)%count
141 is = check%recv(m)%is(n); ie = check%recv(m)%ie(n)
142 js = check%recv(m)%js(n); je = check%recv(m)%je(n)
143 tMe = check%recv(m)%tileMe(n)
144 select case( check%recv(m)%rotation(n) )
145 case(ZERO)
146 do l = 1, l_size ! loop over number of fields
147 ptr_field = f_addrs(l, tMe)
148 do k = 1,ke
149 do j = js, je
150 do i = is, ie
151 pos = pos + 1
152 buffer(pos) = field(i,j,k)
153 end do
154 end do
155 end do
156 end do
157 case(MINUS_NINETY)
158 do l = 1, l_size ! loop over number of fields
159 ptr_field = f_addrs(l, tMe)
160 do k = 1,ke
161 do j = je, js, -1
162 do i = is, ie
163
164 pos = pos + 1
165 buffer(pos) = field(i,j,k)
166 end do
167 end do
168 end do
169 end do
170 case(NINETY)
171 do l = 1, l_size ! loop over number of fields
172 ptr_field = f_addrs(l, tMe)
173 do k = 1,ke
174 do j = js, je
175 do i = ie, is, -1
176
177 pos = pos + 1
178 buffer(pos) = field(i,j,k)
179 end do
180 end do
181 end do
182 end do
183 case(ONE_HUNDRED_EIGHTY)
184 do l = 1, l_size ! loop over number of fields
185 ptr_field = f_addrs(l, tMe)
186 do k = 1,ke
187 do j = je, js, -1
188 do i = ie, is, -1
189 pos = pos + 1
190 buffer(pos) = field(i,j,k)
191 end do
192 end do
193 end do
194 end do
195 end select
196 end do
197 msgsize = pos - buffer_pos
198 if( msgsize.GT.0 )then
199 to_pe = check%recv(m)%pe
200 mpp_domains_stack_hwm = max( mpp_domains_stack_hwm, pos)
201 if( mpp_domains_stack_hwm.GT.mpp_domains_stack_size )then
202 write( text,'(i8)' )mpp_domains_stack_hwm
203 call mpp_error( FATAL, 'MPP_DO_CHECK: mpp_domains_stack overflow, ' // &
204 'call mpp_domains_set_stack_size('//trim(text)//') from all PEs.')
205 end if
206 call mpp_send( buffer(buffer_pos+1), plen=msgsize, to_pe=to_pe, tag=COMM_TAG_2 )
207 buffer_pos = pos
208 end if
209 end do ! end do list = 0,nlist-1
210
211 call mpp_sync_self(check=EVENT_RECV) ! To ensure recv is completed.
212 buffer_pos = buffer_recv_size
213 !--- compare the data in reverse order
214 CHECK_LOOP: do m = check%nrecv, 1, -1
215 do n = check%recv(m)%count, 1, -1
216 is = check%recv(m)%is(n); ie = check%recv(m)%ie(n)
217 js = check%recv(m)%js(n); je = check%recv(m)%je(n)
218 msgsize = (ie-is+1)*(je-js+1)*ke*l_size
219 pos = buffer_pos - msgsize
220 buffer_pos = pos
221 tMe = check%recv(m)%tileMe(n)
222 do l=1, l_size ! loop over number of fields
223 ptr_field = f_addrs(l, tMe)
224 do k = 1,ke
225 do j = js, je
226 do i = is, ie
227 pos = pos + 1
228 if( field(i,j,k) .NE. buffer(pos) ) then
229 print*,"Error from MPP_DO_CHECK on pe = ", mpp_pe(), ": field ", &
230 trim(field_name), " at point (", i, ",", j, ",", k, ") = ", field(i,j,k), &
231 " does not equal to the value = ", buffer(pos), " on pe ", check%recv(m)%pe
232 call mpp_error(debug_update_level, "MPP_DO_CHECK: mismatch on the boundary for symmetry point")
233 exit CHECK_LOOP
234 end if
235 end do
236 end do
237 end do
238 end do
239 end do
240 end do CHECK_LOOP ! end do list = nlist-1,0,-1
241 call mpp_sync_self()
242
243 return
244 end subroutine MPP_DO_CHECK_3D_

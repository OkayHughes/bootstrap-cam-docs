25 subroutine MPP_TRANSMIT_( put_data, put_len, to_pe, get_data, get_len, from_pe, block, tag, recv_request )
26!a message-passing routine intended to be reminiscent equally of both MPI and SHMEM
27
28!put_data and get_data are contiguous MPP_TYPE_ arrays
29
30!at each call, your put_data array is put to to_pe's get_data
31! your get_data array is got from from_pe's put_data
32!i.e we assume that typically (e.g updating halo regions) each PE performs a put _and_ a get
33
34!special PE designations:
35! NULL_PE: to disable a put or a get (e.g at boundaries)
36! ANY_PE: if remote PE for the put or get is to be unspecific
37! ALL_PES: broadcast and collect operations (collect not yet implemented)
38
39!ideally we would not pass length, but this f77-style call performs better (arrays passed by address, not descriptor)
40!further, this permits <length> contiguous words from an array of any rank to be passed (avoiding f90 rank conformance check)
41
42!caller is responsible for completion checks (mpp_sync_self) before and after
43
44 integer, intent(in) :: put_len, to_pe, get_len, from_pe
45 MPP_TYPE_, intent(in) :: put_data(*)
46 MPP_TYPE_, intent(out) :: get_data(*)
47 logical, intent(in), optional :: block
48 integer, intent(in), optional :: tag
49 integer, intent(out), optional :: recv_request
50
51#include <mpp/shmem.fh>
52 external shmem_ptr
53
54 integer :: i, outunit, errunit
55 integer :: np
56 integer(LONG_KIND) :: data_loc
57!pointer to remote data
58 MPP_TYPE_ :: remote_data(get_len)
59 pointer( ptr_remote_data, remote_data )
60 MPP_TYPE_ :: broadcast_data(get_len)
61 pointer( ptr, broadcast_data )
62 integer :: words
63 character(len=8) :: text
64 MPP_TYPE_, allocatable, save :: local_data(:) !local copy used by non-parallel code (no SHMEM or MPI)
65
66 if( .NOT.module_is_initialized )call mpp_error( FATAL, 'MPP_TRANSMIT: You must first call mpp_init.' )
67 if( to_pe.EQ.NULL_PE .AND. from_pe.EQ.NULL_PE )return
68
69 errunit = stderr()
70 outunit = stdout()
71 if( debug )then
72 call SYSTEM_CLOCK(tick)
73 write( outunit,'(a,i18,a,i5,a,2i5,2i8)' )&
74 'T=',tick, ' PE=',pe, ' MPP_TRANSMIT begin: to_pe, from_pe, put_len, get_len=', to_pe, from_pe, put_len, get_len
75 end if
76
77!do put first and then get
78 if( to_pe.GE.0 .AND. to_pe.LT.npes )then
79!send data pointer to to_pe
80#ifdef _CRAYT90
81 call SHMEM_UDCFLUSH !invalidate data cache
82#endif
83 if( debug .and. (current_clock.NE.0) )call SYSTEM_CLOCK(start_tick)
84 call SHMEM_INT8_WAIT( status(to_pe), MPP_WAIT )
85 status(to_pe) = MPP_WAIT !prohibit puts to to_pe until it has retrieved this message
86 if( debug .and. (current_clock.NE.0) )call increment_current_clock(EVENT_WAIT)
87#ifdef __ia64
88 data_loc = shmem_ptr(put_data,pe)
89! write(0,*)'pe, data_loc, loc(put_data)=', pe, data_loc, loc(put_data)
90#else
91 data_loc = LOC(put_data)
92#endif
93 if( debug .and. (current_clock.NE.0) )call SYSTEM_CLOCK(start_tick)
94 call SHMEM_INTEGER_PUT( mpp_from_pe, pe, 1, to_pe )
95 call SHMEM_PUT8( remote_data_loc(pe), data_loc, 1, to_pe )
96 if( debug .and. (current_clock.NE.0) )call increment_current_clock( EVENT_SEND, put_len*MPP_TYPE_BYTELEN_ )
97 else if( to_pe.EQ.ALL_PES )then !this is a broadcast from from_pe
98 if( from_pe.LT.0 .OR. from_pe.GE.npes )call mpp_error( FATAL, 'MPP_TRANSMIT: broadcasting from invalid PE.' )
99 if( put_len.GT.get_len )call mpp_error( FATAL, 'MPP_TRANSMIT: size mismatch between put_data and get_data.' )
100 if( pe.EQ.from_pe )then
101 if( LOC(get_data).NE.LOC(put_data) )then
102!dir$ IVDEP
103 do i = 1,get_len
104 get_data(i) = put_data(i)
105 end do
106 end if
107 end if
108 call mpp_broadcast( get_data, get_len, from_pe )
109 return
110
111 else if( to_pe.EQ.ANY_PE )then !we don't have a destination to do puts to, so only do gets
112 if( from_pe.LT.0 .OR. from_pe.GE.npes )call mpp_error( FATAL, 'MPP_TRANSMIT: invalid from_pe along with to_pe=ANY_PE.' )
113 if( debug .and. (current_clock.NE.0) )call SYSTEM_CLOCK(start_tick)
114 call SHMEM_GET_( get_data, put_data, get_len, from_pe )
115 call SHMEM_PUT8( status(pe), MPP_READY, 1, from_pe ) !tell from_pe that you have retrieved this message
116 if( debug .and. (current_clock.NE.0) )call increment_current_clock( EVENT_RECV, get_len*MPP_TYPE_BYTELEN_ )
117 return
118 else if( to_pe.NE.NULL_PE )then !no other valid cases except NULL_PE
119 call mpp_error( FATAL, 'MPP_TRANSMIT: invalid to_pe.' )
120 end if
121
122!do the get: for libSMA, a get means do a wait to ensure put on remote PE is complete
123 if( from_pe.GE.0 .AND. from_pe.LT.npes )then
124#ifdef _CRAYT90
125 call SHMEM_UDCFLUSH !invalidate data cache
126#endif
127 if( debug .and. (current_clock.NE.0) )call SYSTEM_CLOCK(start_tick)
128 if( debug )write( errunit,* )'pe, from_pe, remote_data_loc(from_pe)=', pe, from_pe, remote_data_loc(from_pe)
129 call SHMEM_INT8_WAIT( remote_data_loc(from_pe), MPP_WAIT )
130 if( debug .and. (current_clock.NE.0) )call increment_current_clock(EVENT_WAIT)
131 ptr_remote_data = remote_data_loc(from_pe)
132 remote_data_loc(from_pe) = MPP_WAIT !reset
133! call SHMEM_PUT8( status(pe), MPP_READY, 1, from_pe ) !tell from_pe we have retrieved the location
134 if( debug .and. (current_clock.NE.0) )call SYSTEM_CLOCK(start_tick)
135#if defined(CRAYPVP) || defined(sgi_mipspro) || defined(__ia64)
136!since we have the pointer to remote data, just retrieve it with a simple copy
137 if( LOC(get_data).NE.LOC(remote_data) )then
138!dir$ IVDEP
139 do i = 1,get_len
140 get_data(i) = remote_data(i)
141 end do
142 else
143 call mpp_error(FATAL)
144 end if
145#else
146 call SHMEM_GET_( get_data, remote_data, get_len, from_pe )
147#endif
148 call SHMEM_PUT8( status(pe), MPP_READY, 1, from_pe ) !tell from_pe we have retrieved the location
149 if( debug .and. (current_clock.NE.0) )call increment_current_clock( EVENT_RECV, get_len*MPP_TYPE_BYTELEN_ )
150
151 else if( from_pe.EQ.ANY_PE )then
152#ifdef _CRAYT90
153 call SHMEM_UDCFLUSH !invalidate data cache
154#endif
155!since we don't know which PE is sending us data, we wait for remote PE to send us its ID
156!this is only required for !CRAYPVP && !sgi_mipspro, but is done there too, so that we can send put_is_done back.
157 call shmem_integer_wait( mpp_from_pe, ANY_PE )
158 if( debug .and. (current_clock.NE.0) )call SYSTEM_CLOCK(start_tick)
159 call SHMEM_INT8_WAIT( remote_data_loc(mpp_from_pe), MPP_WAIT )
160 if( debug .and. (current_clock.NE.0) )call increment_current_clock(EVENT_WAIT)
161 ptr_remote_data = remote_data_loc(mpp_from_pe)
162 remote_data_loc(mpp_from_pe) = MPP_WAIT !reset
163 call SHMEM_PUT8( status(pe), MPP_READY, 1, mpp_from_pe ) !tell mpp_from_pe we have retrieved the location
164#if defined(CRAYPVP) || defined(sgi_mipspro) || defined(__ia64)
165!since we have the pointer to remote data, just retrieve it with a simple copy
166 if( debug .and. (current_clock.NE.0) )call SYSTEM_CLOCK(start_tick)
167 if( LOC(get_data).NE.LOC(remote_data) )then
168!dir$ IVDEP
169 do i = 1,get_len
170 get_data(i) = remote_data(i)
171 end do
172 end if
173#else
174 call SHMEM_GET_( get_data, remote_data, get_len, mpp_from_pe )
175#endif
176 if( debug .and. (current_clock.NE.0) )call increment_current_clock( EVENT_RECV, get_len*MPP_TYPE_BYTELEN_ )
177 mpp_from_pe = ANY_PE !reset
178 else if( from_pe.EQ.ALL_PES )then
179 call mpp_error( FATAL, 'MPP_TRANSMIT: from_pe=ALL_PES has ambiguous meaning, and hence is not implemented.' )
180
181 else if( from_pe.NE.NULL_PE )then !only remaining valid choice is NULL_PE
182 call mpp_error( FATAL, 'MPP_TRANSMIT: invalid from_pe.' )
183 end if
184
185 if( debug )then
186 call SYSTEM_CLOCK(tick)
187 write( outunit,'(a,i18,a,i5,a,2i5,2i8)' )&
188 'T=',tick, ' PE=',pe, ' MPP_TRANSMIT end: to_pe, from_pe, put_len, get_len=', to_pe, from_pe, put_len, get_len
189 end if
190 return
191 end subroutine MPP_TRANSMIT_

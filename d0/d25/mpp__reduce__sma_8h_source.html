19 subroutine MPP_REDUCE_0D_( a, pelist )
20!find the max of scalar a the PEs in pelist (all PEs if this argument is omitted)
21!result is also automatically broadcast to all PEs
22 MPP_TYPE_, intent(inout) :: a
23 integer, intent(in), optional :: pelist(0:)
24 integer :: n
25!work holds pWrk array + 1 word for symmetric copy of a
26 MPP_TYPE_ :: work(SHMEM_REDUCE_MIN_WRKDATA_SIZE+1)
27 pointer( ptr, work )
28 integer :: words
29 character(len=8) :: text
30
31 if( .NOT.module_is_initialized )call mpp_error( FATAL, 'MPP_REDUCE_0D: You must first call mpp_init.' )
32 n = get_peset(pelist); if( peset(n)%count.EQ.1 )return
33
34 if( debug .and. (current_clock.NE.0) )call SYSTEM_CLOCK(start_tick)
35!allocate space from the stack for pwrk and b
36 ptr = LOC(mpp_stack)
37 words = size(work(:))*size(transfer(work(1),word))
38 if( words.GT.mpp_stack_size )then
39 write( text, '(i8)' )words
40 call mpp_error( FATAL, 'MPP_REDUCE_0D user stack overflow: call mpp_set_stack_size('//text//') from all PEs.' )
41 end if
42 mpp_stack_hwm = max( words, mpp_stack_hwm )
43
44 work(1) = a
45 call SHMEM_REDUCE_( work, work, 1, peset(n)%start, peset(n)%log2stride, peset(n)%count, work(2), sync )
46 call mpp_sync(pelist)
47 a = work(1)
48 if( debug .and. (current_clock.NE.0) )call increment_current_clock( EVENT_ALLREDUCE, MPP_TYPE_BYTELEN_ )
49 return
50 end subroutine MPP_REDUCE_0D_

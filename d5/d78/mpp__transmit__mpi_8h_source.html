25 subroutine MPP_TRANSMIT_( put_data, put_len, to_pe, get_data, get_len, from_pe, block, tag, recv_request, send_request )
26!a message-passing routine intended to be reminiscent equally of both MPI and SHMEM
27
28!put_data and get_data are contiguous MPP_TYPE_ arrays
29
30!at each call, your put_data array is put to to_pe's get_data
31! your get_data array is got from from_pe's put_data
32!i.e we assume that typically (e.g updating halo regions) each PE performs a put _and_ a get
33
34!special PE designations:
35! NULL_PE: to disable a put or a get (e.g at boundaries)
36! ANY_PE: if remote PE for the put or get is to be unspecific
37! ALL_PES: broadcast and collect operations (collect not yet implemented)
38
39!ideally we would not pass length, but this f77-style call performs better (arrays passed by address, not descriptor)
40!further, this permits <length> contiguous words from an array of any rank to be passed (avoiding f90 rank conformance check)
41
42!caller is responsible for completion checks (mpp_sync_self) before and after
43
44 integer, intent(in) :: put_len, to_pe, get_len, from_pe
45 MPP_TYPE_, intent(in) :: put_data(*)
46 MPP_TYPE_, intent(out) :: get_data(*)
47 logical, intent(in), optional :: block
48 integer, intent(in), optional :: tag
49 integer, intent(out), optional :: recv_request, send_request
50 logical :: block_comm
51 integer :: i
52 MPP_TYPE_, allocatable, save :: local_data(:) !local copy used by non-parallel code (no SHMEM or MPI)
53 integer :: comm_tag
54 integer :: rsize
55
56 if( .NOT.module_is_initialized )call mpp_error( FATAL, 'MPP_TRANSMIT: You must first call mpp_init.' )
57 if( to_pe.EQ.NULL_PE .AND. from_pe.EQ.NULL_PE )return
58
59 block_comm = .true.
60 if(PRESENT(block)) block_comm = block
61
62 if( debug )then
63 call SYSTEM_CLOCK(tick)
64 write( stdout_unit,'(a,i18,a,i6,a,2i6,2i8)' )&
65 'T=',tick, ' PE=',pe, ' MPP_TRANSMIT begin: to_pe, from_pe, put_len, get_len=', to_pe, from_pe, put_len, get_len
66 end if
67
68 comm_tag = DEFAULT_TAG
69 if(present(tag)) comm_tag = tag
70
71!do put first and then get
72 if( to_pe.GE.0 .AND. to_pe.LT.npes )then
73!use non-blocking sends
74 if( debug .and. (current_clock.NE.0) )call SYSTEM_CLOCK(start_tick)
75!z1l: truly non-blocking send.
76! if( request(to_pe).NE.MPI_REQUEST_NULL )then !only one message from pe->to_pe in queue
77! if( debug )write( stderr(),* )'PE waiting for sending', pe, to_pe
78! call MPI_WAIT( request(to_pe), stat, error )
79! end if
80 if(present(send_request)) then
81 call MPI_ISEND( put_data, put_len, MPI_TYPE_, to_pe, comm_tag, mpp_comm_private, send_request, error)
82 else
83 cur_send_request = cur_send_request + 1
84 if( cur_send_request > max_request ) call mpp_error(FATAL, &
85 "MPP_TRANSMIT: cur_send_request is greater than max_request, increase mpp_nml request_multiply")
86 call MPI_ISEND( put_data, put_len, MPI_TYPE_, to_pe, comm_tag, mpp_comm_private, request_send(cur_send_request), error)
87 endif
88 if( debug .and. (current_clock.NE.0) )call increment_current_clock( EVENT_SEND, put_len*MPP_TYPE_BYTELEN_ )
89 else if( to_pe.EQ.ALL_PES )then !this is a broadcast from from_pe
90 if( from_pe.LT.0 .OR. from_pe.GE.npes )call mpp_error( FATAL, 'MPP_TRANSMIT: broadcasting from invalid PE.' )
91 if( put_len.GT.get_len )call mpp_error( FATAL, 'MPP_TRANSMIT: size mismatch between put_data and get_data.' )
92 if( pe.EQ.from_pe )then
93 if( LOC(get_data).NE.LOC(put_data) )then
94!dir$ IVDEP
95 do i = 1,get_len
96 get_data(i) = put_data(i)
97 end do
98 end if
99 end if
100 call mpp_broadcast( get_data, get_len, from_pe )
101 return
102 else if( to_pe.EQ.ANY_PE )then !we don't have a destination to do puts to, so only do gets
103!...but you cannot have a pure get with MPI
104 call mpp_error( FATAL, 'MPP_TRANSMIT: you cannot transmit to ANY_PE using MPI.' )
105 else if( to_pe.NE.NULL_PE )then !no other valid cases except NULL_PE
106 call mpp_error( FATAL, 'MPP_TRANSMIT: invalid to_pe.' )
107 end if
108
109!do the get: for libSMA, a get means do a wait to ensure put on remote PE is complete
110 if( from_pe.GE.0 .AND. from_pe.LT.npes )then
111!receive from from_pe
112 if( debug .and. (current_clock.NE.0) )call SYSTEM_CLOCK(start_tick)
113 if( block_comm ) then
114 call MPI_RECV( get_data, get_len, MPI_TYPE_, from_pe, comm_tag, mpp_comm_private, stat, error )
115 call MPI_GET_COUNT( stat, MPI_TYPE_, rsize, error)
116 if(rsize .NE. get_len) then
117 print*, "rsize, get_len=", rsize, get_len, mpp_pe(), from_pe
118 call mpp_error(FATAL, "MPP_TRANSMIT: get_len does not match size of data received")
119 endif
120 else
121! if( request_recv(from_pe).NE.MPI_REQUEST_NULL )then !only one message from from_pe->pe in queue
122 ! if( debug )write( stderr(),* )'PE waiting for receiving', pe, from_pe
123! call MPI_WAIT( request_recv(from_pe), stat, error )
124! end if
125 if(PRESENT(recv_request)) then
126 call MPI_IRECV( get_data, get_len, MPI_TYPE_, from_pe, comm_tag, mpp_comm_private, &
127 recv_request, error )
128 else
129 cur_recv_request = cur_recv_request + 1
130 if( cur_recv_request > max_request ) call mpp_error(FATAL, &
131 "MPP_TRANSMIT: cur_recv_request is greater than max_request, increase mpp_nml request_multiply")
132 call MPI_IRECV( get_data, get_len, MPI_TYPE_, from_pe, comm_tag, mpp_comm_private, &
133 request_recv(cur_recv_request), error )
134 size_recv(cur_recv_request) = get_len
135 type_recv(cur_recv_request) = MPI_TYPE_
136 endif
137 endif
138 if( debug .and. (current_clock.NE.0) )call increment_current_clock( EVENT_RECV, get_len*MPP_TYPE_BYTELEN_ )
139 else if( from_pe.EQ.ANY_PE )then
140!receive from MPI_ANY_SOURCE
141 if( debug .and. (current_clock.NE.0) )call SYSTEM_CLOCK(start_tick)
142 call MPI_RECV( get_data, get_len, MPI_TYPE_, MPI_ANY_SOURCE, comm_tag, mpp_comm_private, stat, error )
143 if( debug .and. (current_clock.NE.0) )call increment_current_clock( EVENT_RECV, get_len*MPP_TYPE_BYTELEN_ )
144 else if( from_pe.EQ.ALL_PES )then
145 call mpp_error( FATAL, 'MPP_TRANSMIT: from_pe=ALL_PES has ambiguous meaning, and hence is not implemented.' )
146 else if( from_pe.NE.NULL_PE )then !only remaining valid choice is NULL_PE
147 call mpp_error( FATAL, 'MPP_TRANSMIT: invalid from_pe.' )
148 end if
149
150 if( debug )then
151 call SYSTEM_CLOCK(tick)
152 write( stdout_unit,'(a,i18,a,i6,a,2i6,2i8)' )&
153 'T=',tick, ' PE=',pe, ' MPP_TRANSMIT end: to_pe, from_pe, put_len, get_len=', to_pe, from_pe, put_len, get_len
154 end if
155 return
156 end subroutine MPP_TRANSMIT_

164 subroutine MPP_BROADCAST_( data, length, from_pe, pelist )
165!this call was originally bundled in with mpp_transmit, but that doesn't allow
166!broadcast to a subset of PEs. This version will, and mpp_transmit will remain
167!backward compatible.
168 MPP_TYPE_, intent(inout) :: data(*)
169 integer, intent(in) :: length, from_pe
170 integer, intent(in), optional :: pelist(:)
171 integer :: n, i, from_rank, stdout_unit
172
173 if( .NOT.module_is_initialized )call mpp_error( FATAL, 'MPP_BROADCAST: You must first call mpp_init.' )
174 n = get_peset(pelist); if( peset(n)%count.EQ.1 )return
175
176 if( debug )then
177 call SYSTEM_CLOCK(tick)
178 write( stdout_unit,'(a,i18,a,i6,a,2i6,2i8)' )&
179 'T=',tick, ' PE=',pe, ' MPP_BROADCAST begin: from_pe, length=', from_pe, length
180 end if
181
182 if( .NOT.ANY(from_pe.EQ.peset(current_peset_num)%list) ) &
183 call mpp_error( FATAL, 'MPP_BROADCAST: broadcasting from invalid PE.' )
184
185 if( debug .and. (current_clock.NE.0) )call SYSTEM_CLOCK(start_tick)
186 ! find the rank of from_pe in the pelist.
187 do i = 1, mpp_npes()
188 if(peset(n)%list(i) == from_pe) then
189 from_rank = i - 1
190 exit
191 endif
192 enddo
193 if( mpp_npes().GT.1 )call MPI_BCAST( data, length, MPI_TYPE_, from_rank, peset(n)%id, error )
194 if( debug .and. (current_clock.NE.0) )call increment_current_clock( EVENT_BROADCAST, length*MPP_TYPE_BYTELEN_ )
195 return
196 end subroutine MPP_BROADCAST_

19subroutine MPP_GATHER_1D_(sbuf, rbuf,pelist)
20! JWD: Did not create mpp_gather_2d because have no requirement for it
21! JWD: See mpp_gather_2dv below
22 MPP_TYPE_, dimension(:), intent(in) :: sbuf
23 MPP_TYPE_, dimension(:), intent(inout) :: rbuf
24 integer, dimension(:), intent(in), optional :: pelist(:)
25
26 integer :: cnt, l, nproc, op_root
27 integer, allocatable :: pelist2(:)
28
29
30! If pelist is provided, the first position must be
31! the operation root
32 if(PRESENT(pelist))then
33 nproc = size(pelist)
34 allocate(pelist2(nproc))
35 pelist2 = pelist
36 else
37 nproc = mpp_npes()
38 allocate(pelist2(nproc))
39 pelist2 = (/ (l, l=root_pe, nproc-1+root_pe) /)
40 endif
41 op_root = pelist2(1)
42
43 cnt = size(sbuf(:))
44 if(size(rbuf(:)) < cnt*nproc) call mpp_error(FATAL, &
45 "MPP_GATHER_1D_: size(rbuf) must be at least npes*size(sbuf) ")
46
47 !--- pre-post receiving
48 if(pe == op_root) then
49 rbuf(1:cnt) = sbuf
50 do l = 2, nproc
51 call mpp_recv(rbuf((l-1)*cnt+1), glen=cnt, from_pe=pelist2(l), block=.FALSE., tag=COMM_TAG_1 )
52 enddo
53 else
54 call mpp_send(sbuf(1), plen=cnt, to_pe=op_root, tag=COMM_TAG_1)
55 endif
56
57 call mpp_sync_self(check=EVENT_RECV)
58 call mpp_sync_self()
59 deallocate(pelist2)
60end subroutine MPP_GATHER_1D_

62subroutine MPP_GATHER_1DV_(sbuf, ssize, rbuf, rsize, pelist)
63 MPP_TYPE_, dimension(:), intent(in) :: sbuf
64 MPP_TYPE_, dimension(:), intent(inout) :: rbuf
65 integer, intent(in) :: ssize
66 integer, dimension(:), intent(in) :: rsize
67 integer, dimension(:), intent(in), optional :: pelist(:)
68
69 integer :: cnt, l, nproc, pos, op_root
70 integer, allocatable :: pelist2(:)
71
72! If pelist is provided, the first position must be
73! the operation root
74 if(PRESENT(pelist))then
75 nproc = size(pelist)
76 allocate(pelist2(nproc))
77 pelist2 = pelist
78 else
79 nproc = mpp_npes()
80 allocate(pelist2(nproc))
81 pelist2 = (/ (l, l=0+root_pe, nproc-1+root_pe) /)
82 endif
83 op_root = pelist2(1)
84
85
86 !--- pre-post receiving
87 if (pe .eq. op_root) then
88 pos = 1
89 do l = 1,nproc ! include op_root to simplify logic
90 if (rsize(l) == 0) then
91 cycle ! avoid ranks with no data
92 endif
93 call mpp_recv(rbuf(pos),glen=rsize(l),from_pe=pelist2(l), &
94 block=.FALSE.,tag=COMM_TAG_2)
95 pos = pos + rsize(l)
96 enddo
97 endif
98 if (ssize .gt. 0) then
99 call mpp_send(sbuf(1),plen=ssize,to_pe=op_root,tag=COMM_TAG_2) !avoid ranks with no data
100 endif
101
102 call mpp_sync_self(check=EVENT_RECV)
103 call mpp_sync_self()
104 deallocate(pelist2)
105end subroutine MPP_GATHER_1DV_

131subroutine MPP_GATHER_PELIST_3D_(is, ie, js, je, nk, pelist, array_seg, data, is_root_pe, &
132 ishift, jshift)
133 integer, intent(in) :: is, ie, js, je, nk
134 integer, dimension(:), intent(in) :: pelist
135 MPP_TYPE_, dimension(is:ie,js:je,1:nk), intent(in) :: array_seg
136 MPP_TYPE_, dimension(:,:,:), intent(inout) :: data
137 logical, intent(in) :: is_root_pe
138 integer, optional, intent(in) :: ishift, jshift
139
140 integer :: i, msgsize, root_pe, root_pe_test
141 integer :: i1, i2, j1, j2, ioff, joff
142 integer :: my_ind(4), gind(4,size(pelist))
143 type array3D
144 MPP_TYPE_, dimension(:,:,:), allocatable :: data
145 endtype array3D
146 type(array3d), dimension(:), allocatable :: temp
147
148 if (.not.ANY(mpp_pe().eq.pelist(:))) return
149
150 if (is_root_pe) then
151 root_pe = mpp_pe()
152 root_pe_test = 999
153 if (.not.ANY(pelist(:).eq.root_pe)) call mpp_error(FATAL, &
154 "fms_io(mpp_gather_pelist): root_pe not a member of pelist")
155 else
156 root_pe = 0
157 root_pe_test = -999
158 endif
159! need this check in case MPI-rank 0 is a member of the pelist
160 call mpp_max(root_pe_test, pelist)
161 if (root_pe_test.lt.0) call mpp_error(FATAL, &
162 "fms_io(mpp_gather_pelist): root_pe not specified or not a member of the pelist")
163! need to make sure only one root_pe has been specified
164 call mpp_sum(root_pe, pelist)
165 if ((is_root_pe) .and. (mpp_pe().ne.root_pe)) call mpp_error(FATAL, &
166 "fms_io(mpp_gather_pelist): too many root_pes specified")
167
168
169 ioff=0
170 joff=0
171 if (present(ishift)) ioff=ishift
172 if (present(jshift)) joff=jshift
173
174 my_ind(1) = is
175 my_ind(2) = ie
176 my_ind(3) = js
177 my_ind(4) = je
178
179! gather indices into global index on root_pe
180 if (is_root_pe) then
181 allocate(temp(1:size(pelist)))
182 do i = 1, size(pelist)
183! root_pe data copy - no send to self
184 if (pelist(i).eq.root_pe) then
185 gind(:,i) = my_ind(:)
186 else
187 call mpp_recv(gind(:,i:i), 4, pelist(i), .FALSE., COMM_TAG_1)
188 endif
189 enddo
190 call mpp_sync_self(check=EVENT_RECV)
191 gind(1,:)=gind(1,:)+ioff
192 gind(2,:)=gind(2,:)+ioff
193 gind(3,:)=gind(3,:)+joff
194 gind(4,:)=gind(4,:)+joff
195! check indices to make sure they are within the range of "data"
196 if ((minval(gind).lt.1) .OR. (maxval(gind(1:2,:)).gt.size(data,1)) .OR. (maxval(gind(3:4,:)).gt.size(data,2))) &
197 call mpp_error(FATAL,"fms_io(mpp_gather_pelist): specified indices (with shift) are outside of the &
198 &range of the receiving array")
199 else
200! non root_pe's send indices to root_pe
201 call mpp_send(my_ind(:), 4, root_pe, COMM_TAG_1)
202 call mpp_sync_self(check=EVENT_SEND)
203 endif
204
205! gather segments into data based on indices
206 if (is_root_pe) then
207 do i = 1, size(pelist)
208 if (pelist(i).ne.root_pe) then ! no send to self
209 i1 = gind(1,i)
210 i2 = gind(2,i)
211 j1 = gind(3,i)
212 j2 = gind(4,i)
213 msgsize = (i2-i1+1)*(j2-j1+1)*nk
214 allocate(temp(i)%data(i1:i2,j1:j2,1:nk))
215 call mpp_recv(temp(i)%data(i1:i2,j1:j2,1:nk), msgsize, pelist(i), .FALSE., COMM_TAG_2)
216 endif
217 enddo
218 call mpp_sync_self(check=EVENT_RECV)
219! unbuffer/copy the data into the return array
220 do i = 1, size(pelist)
221 if (pelist(i).eq.root_pe) then
222! data copy - no send to self
223 data(is+ioff:ie+ioff,js+joff:je+joff,1:nk) = array_seg(is:ie,js:je,1:nk)
224 else
225 i1 = gind(1,i)
226 i2 = gind(2,i)
227 j1 = gind(3,i)
228 j2 = gind(4,i)
229 data(i1:i2,j1:j2,1:nk)=temp(i)%data(i1:i2,j1:j2,1:nk)
230 deallocate(temp(i)%data)
231 endif
232 enddo
233 deallocate(temp)
234 else
235! non root_pe's send data to root_pe
236 msgsize = (my_ind(2)-my_ind(1)+1) * (my_ind(4)-my_ind(3)+1) * nk
237 call mpp_send(array_seg, msgsize, root_pe, COMM_TAG_2)
238 call mpp_sync_self(check=EVENT_SEND)
239 endif
240
241 call mpp_sync_self()
242 return

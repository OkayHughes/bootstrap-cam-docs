38 nlist = size(domain%list(:))
39 tile = 1
40 if(present(tile_count)) tile = tile_count
41 global_flag = NON_BITWISE_EXACT_SUM
42 if(present(flags)) global_flag = flags

44 call mpp_get_domain_shift(domain, ishift, jshift, position)

46 if( size(field,1).EQ.domain%x(tile)%compute%size+ishift .AND. size(field,2).EQ.domain%y(tile)%compute%size+jshift )then
47!field is on compute domain
48 ioff = -domain%x(tile)%compute%begin + 1
49 joff = -domain%y(tile)%compute%begin + 1
50 else if( size(field,1).EQ.domain%x(tile)%memory%size+ishift .AND. size(field,2).EQ.domain%y(tile)%memory%size+jshift )then
51!field is on data domain
52 ioff = -domain%x(tile)%data%begin + 1
53 joff = -domain%y(tile)%data%begin + 1
54 else

62 isc = is; iec = ie + ishift; jsc = js; jec = je + jshift
63
64 call mpp_get_global_domain(domain, xsize = gxsize, ysize = gysize )
65 MPP_GLOBAL_SUM_ = 0

73 end do
74 end do
75 allocate( global2D( gxsize+ishift, gysize+jshift ) )
76 global2D = 0.

78 !call mpp_global_field( domain, field2D, global2D, position=position, tile_count=tile_count )
79
80 if ( present( tile_count ) ) then

82 else
83 call mpp_global_field( domain, field2D, global2D, position=position )
84 endif
85
86 ioffset = domain%x(tile)%goffset*ishift; joffset = domain%y(tile)%goffset*jshift
87 mygsum(tile) = sum(global2D(1:gxsize+ioffset,1:gysize+joffset))
88 deallocate(global2D, field2d)
89 if( tile == ntile) then
90 if(domain%ntiles == 1 ) then
91 MPP_GLOBAL_SUM_ = mygsum(tile)
92 else if( nlist == 1) then
93 MPP_GLOBAL_SUM_ = sum(mygsum(1:ntile))
94 else ! need to sum by the order of tile_count
95 ! first fill the global sum on current pe.
96 do n = 1, ntile

113 gsum(domain%list(m)%tile_id(n)) = nbrgsum(n)
114 end do
115 end if
116 end do
117
118 MPP_GLOBAL_SUM_ = sum(gsum(1:domain%ntiles))
119 end if
120 end if
121 else if ( global_flag == BITWISE_EFP_SUM )then
122#ifdef DO_EFP_SUM_
123 !this is bitwise across different PE counts using EFP sum
124 if( ntile > 1 ) then
125 call mpp_error( FATAL, 'MPP_GLOBAL_SUM_: multiple tile per pe is not supported for BITWISE_EFP_SUM')
126 endif
127 allocate( field2D (isc:iec,jsc:jec) )
128 do j = jsc, jec
129 do i = isc, iec
130 field2D(i,j) = sum( field(i+ioff:i+ioff,j+joff:j+joff MPP_EXTRA_INDICES_) )
131 end do
132 end do
133 !--- using efp sum.
134 if(efp_sum_overflow_check) then
135 MPP_GLOBAL_SUM_ = mpp_reproducing_sum(field2D, overflow_check=.true.)
136 else
137 MPP_GLOBAL_SUM_ = mpp_reproducing_sum(field2D, overflow_check=overflow_check)
138 endif
139#else
140 call mpp_error( FATAL, 'MPP_GLOBAL_SUM_: BITWISE_EFP_SUM is only implemented for real number, contact developer')
141#endif
142 else !this is not bitwise-exact across different PE counts
143 ioffset = domain%x(tile)%loffset*ishift; joffset = domain%y(tile)%loffset*jshift
144 mygsum(tile) = sum( field(is+ioff:ie+ioff+ioffset, js+joff:je+joff+joffset MPP_EXTRA_INDICES_) )
145 if(tile == ntile) then
146 MPP_GLOBAL_SUM_ = sum(mygsum(1:ntile))
147 call mpp_sum( MPP_GLOBAL_SUM_, domain%list(:)%pe )
148 end if
149 end if
150
151 return
152 end function MPP_GLOBAL_SUM_

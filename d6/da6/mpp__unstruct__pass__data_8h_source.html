37 type(domainUG), intent(in) :: UG_domain
38 MPP_TYPE_, intent(inout) :: field_UG(:,:)
39 MPP_TYPE_, intent(in) :: field_SG(:,:,:)
40 MPP_TYPE_ :: buffer(mpp_domains_stack_size)
41 character(len=8) :: text
42 integer :: i, j, k, l, m, ke, from_pe, to_pe
43 integer :: buffer_pos, pos, msgsize, ioff, joff
44
45 !--- check if data is on data or computing domain
46 if(size(field_SG,1) .EQ. UG_domain%SG_domain%x(1)%compute%size .AND. &
47 size(field_SG,2) .EQ. UG_domain%SG_domain%y(1)%compute%size) then
48 ioff = 1 - UG_domain%SG_domain%x(1)%compute%begin
49 joff = 1 - UG_domain%SG_domain%y(1)%compute%begin
50 else if(size(field_SG,1) .EQ. UG_domain%SG_domain%x(1)%data%size .AND. &
51 size(field_SG,2) .EQ. UG_domain%SG_domain%y(1)%data%size) then
52 ioff = 1 - UG_domain%SG_domain%x(1)%data%begin
53 joff = 1 - UG_domain%SG_domain%y(1)%data%begin
54 else
55 call mpp_error( FATAL, 'mpp_pass_SG_to_UG_3D_: field_SG must match either compute domain or data domain.' )
56 endif
57
58 ke = size(field_SG,3)
59 buffer_pos = 0
60 !---pre-post receive
61 do m = 1, UG_domain%SG2UG%nrecv
62 msgsize = UG_domain%SG2UG%recv(m)%count*ke
63 if( msgsize.GT.0 )then
64 from_pe = UG_domain%SG2UG%recv(m)%pe
65 mpp_domains_stack_hwm = max( mpp_domains_stack_hwm, (buffer_pos+msgsize) )
66 if( mpp_domains_stack_hwm.GT.mpp_domains_stack_size )then
67 write( text,'(i8)' )mpp_domains_stack_hwm
68 call mpp_error( FATAL, 'mpp_pass_SG_to_UG_3D: mpp_domains_stack overflow, '// &
69 'call mpp_domains_set_stack_size('//trim(text)//') from all PEs.' )
70 end if
71 call mpp_recv( buffer(buffer_pos+1), glen=msgsize, from_pe=from_pe, block=.FALSE., tag=COMM_TAG_1 )
72 buffer_pos = buffer_pos + msgsize
73 end if
74 end do
75
76 !---pack and send data
77 do m = 1, UG_domain%SG2UG%nsend
78 pos = buffer_pos
79 msgsize = UG_domain%SG2UG%send(m)%count * ke
80 if( msgsize.GT.0 )then
81 mpp_domains_stack_hwm = max( mpp_domains_stack_hwm, (buffer_pos+msgsize) )
82 if( mpp_domains_stack_hwm.GT.mpp_domains_stack_size )then
83 write( text,'(i8)' )mpp_domains_stack_hwm
84 call mpp_error( FATAL, 'mpp_pass_SG_to_UG_3D: mpp_domains_stack overflow, '// &
85 'call mpp_domains_set_stack_size('//trim(text)//') from all PEs.' )
86 end if
87
88 do k = 1, ke
89 do l = 1, UG_domain%SG2UG%send(m)%count
90 i = UG_domain%SG2UG%send(m)%i(l)+ioff
91 j = UG_domain%SG2UG%send(m)%j(l)+joff
92 pos = pos+1
93 buffer(pos) = field_SG(i,j,k)
94 end do
95 end do
96 to_pe = UG_domain%SG2UG%send(m)%pe
97 call mpp_send( buffer(buffer_pos+1), plen=msgsize, to_pe=to_pe, tag=COMM_TAG_1 )
98 buffer_pos = buffer_pos + msgsize
99 end if
100 end do
101
102 call mpp_sync_self(check=EVENT_RECV)
103 !--- unpack the buffer
104 buffer_pos = 0
105 do m = 1, UG_domain%SG2UG%nrecv
106 pos = buffer_pos
107 do k = 1, ke
108 do l = 1, UG_domain%SG2UG%recv(m)%count
109 pos = pos+1
110 i = UG_domain%SG2UG%recv(m)%i(l)
111 field_UG(i,k) = buffer(pos)
112 enddo
113 enddo
114 buffer_pos = pos
115 enddo
116
117 call mpp_sync_self( )

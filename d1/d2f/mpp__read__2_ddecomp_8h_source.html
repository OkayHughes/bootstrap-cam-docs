116 subroutine READ_RECORD_( unit, field, nwords, data, time_level, domain, position, tile_count, start_in, axsiz_in )
117!routine that is finally called by all mpp_read routines to perform the read
118!a non-netCDF record contains:
119! field ID
120! a set of 4 coordinates (is:ie,js:je) giving the data subdomain
121! a timelevel and a timestamp (=NULLTIME if field is static)
122! 3D real data (stored as 1D)
123!if you are using direct access I/O, the RECL argument to OPEN must be large enough for the above
124!in a global direct access file, record position on PE is given by %record.
125
126!Treatment of timestamp:
127! We assume that static fields have been passed without a timestamp.
128! Here that is converted into a timestamp of NULLTIME.
129! For non-netCDF fields, field is treated no differently, but is written
130! with a timestamp of NULLTIME. There is no check in the code to prevent
131! the user from repeatedly writing a static field.
132
133 integer, intent(in) :: unit, nwords
134 type(fieldtype), intent(in) :: field
135 MPP_TYPE_, intent(inout) :: data(nwords)
136 integer, intent(in), optional :: time_level
137 type(domain2D), intent(in), optional :: domain
138 integer, intent(in), optional :: position, tile_count
139 integer, intent(in), optional :: start_in(:), axsiz_in(:)
140 integer, dimension(size(field%axes(:))) :: start, axsiz
141
142 integer :: tlevel !,subdomain(4)
143
144
145 integer :: i, error, is, ie, js, je, isg, ieg, jsg, jeg
146 type(domain2d), pointer :: io_domain=>NULL()
147
148 if (.not.PRESENT(time_level)) then
149 tlevel = 0
150 else
151 tlevel = time_level
152 endif
153
154 if( .NOT.module_is_initialized )call mpp_error( FATAL, 'READ_RECORD: must first call mpp_io_init.' )
155 if( .NOT.mpp_file(unit)%opened )call mpp_error( FATAL, 'READ_RECORD: invalid unit number.' )
156 if( .NOT.mpp_file(unit)%read_on_this_pe )return
157
158 if( .NOT.mpp_file(unit)%initialized ) call mpp_error( FATAL, 'MPP_READ: must first call mpp_read_meta.' )
159 if( mpp_file(unit)%format .NE. MPP_NETCDF ) call mpp_error( FATAL, 'Currently dont support non-NetCDF mpp read' )
160
161 if (.not.PRESENT(time_level)) then
162 tlevel = 0
163 else
164 tlevel = time_level
165 endif
166
167 if( verbose )print '(a,2i6,2i5)', 'MPP_READ: PE, unit, %id, %time_level =',&
168 pe, unit, mpp_file(unit)%id, tlevel
169 if( PRESENT(start_in) .AND. PRESENT(axsiz_in) ) then
170 if(size(start(:)) > size(start_in(:)) )call mpp_error( FATAL, 'MPP_READ: size(start_in) < size(start)')
171 if(size(axsiz(:)) > size(axsiz_in(:)) )call mpp_error( FATAL, 'MPP_READ: size(axsiz_in) < size(axsiz)')
172 start(:) = start_in(1:size(start(:)))
173 axsiz(:) = axsiz_in(1:size(axsiz(:)))
174 else
175!define netCDF data block to be read:
176! time axis: START = time level
177! AXSIZ = 1
178! space axis: if there is no domain info
179! START = 1
180! AXSIZ = field%size(axis)
181! if there IS domain info:
182! start of domain is compute%start_index for multi-file I/O
183! global%start_index for all other cases
184! this number must be converted to 1 for NF_GET_VAR
185! (netCDF fortran calls are with reference to 1),
186! So, START = compute%start_index - <start of domain> + 1
187! AXSIZ = usually compute%size
188! However, if compute%start_index-compute%end_index+1.NE.compute%size,
189! we assume that the call is passing a subdomain.
190! To pass a subdomain, you must pass a domain2D object that satisfies the following:
191! global%start_index must contain the <start of domain> as defined above;
192! the data domain and compute domain must refer to the subdomain being passed.
193! In this case, START = compute%start_index - <start of domain> + 1
194! AXSIZ = compute%start_index - compute%end_index + 1
195! NOTE: passing of subdomains will fail for multi-PE single-threaded I/O,
196! since that attempts to gather all data on PE 0.
197 start = 1
198 do i = 1,size(field%axes(:))
199 axsiz(i) = field%size(i)
200 if( i .EQ. field%time_axis_index )start(i) = tlevel
201 end do
202 if( PRESENT(domain) )then
203 call mpp_get_compute_domain( domain, is, ie, js, je, tile_count=tile_count, position=position )
204 call mpp_get_global_domain ( domain, isg, ieg, jsg, jeg, tile_count=tile_count, position=position )
205 axsiz(1) = ie-is+1
206 axsiz(2) = je-js+1
207 if( mpp_file(unit)%fileset.EQ.MPP_SINGLE )then
208 if( npes.GT.1 )then
209 start(1) = is - isg + 1
210 start(2) = js - jsg + 1
211 else !--- z1l fix a problem related obc when npes = 1
212 if( ie-is+1.NE.ieg-isg+1 )then
213 start(1) = is - isg + 1
214 axsiz(1) = ie - is + 1
215 end if
216 if( je-js+1.NE.jeg-jsg+1 )then
217 start(2) = js - jsg + 1
218 axsiz(2) = je - js + 1
219 end if
220 end if
221 else if( mpp_file(unit)%io_domain_exist ) then
222 io_domain=>mpp_get_io_domain(domain)
223 call mpp_get_compute_domain( io_domain, is, ie, js, je, tile_count=tile_count, position=position )
224 call mpp_get_global_domain ( io_domain, isg, ieg, jsg, jeg, tile_count=tile_count, position=position )
225 start(1) = is - isg + 1
226 start(2) = js - jsg + 1
227 io_domain => NULL()
228 end if
229 end if
230 endif
231 if( verbose )print '(a,2i6,i6,12i4)', 'READ_RECORD: PE, unit, nwords, start, axsiz=', pe, unit, nwords, start, axsiz
232
233 call READ_RECORD_CORE_(unit, field, nwords, data, start, axsiz)
234
235 return
236 end subroutine READ_RECORD_

251 subroutine MPP_READ_2DDECOMP_3D_( unit, field, domain, data, tindex, tile_count )
252!mpp_read reads <data> which has the domain decomposition <domain>
253 integer, intent(in) :: unit
254 type(fieldtype), intent(in) :: field
255 type(domain2D), intent(in) :: domain
256 MPP_TYPE_, intent(inout) :: data(:,:,:)
257 integer, intent(in), optional :: tindex, tile_count
258
259 MPP_TYPE_, allocatable :: cdata(:,:,:)
260 MPP_TYPE_, allocatable :: gdata(:)
261 integer :: len, lenx,leny,lenz,i,j,k,n
262!NEW: data may be on compute OR data domain
263 logical :: data_has_halos, halos_are_global, x_is_global, y_is_global
264 integer :: is, ie, js, je, isd, ied, jsd, jed, isg, ieg, jsg, jeg, ism, iem, jsm, jem
265 integer :: ioff, joff, position
266
267 call mpp_clock_begin(mpp_read_clock)
268
269 if (.NOT. present(tindex) .AND. mpp_file(unit)%time_level .ne. -1) &
270 call mpp_error(FATAL, 'MPP_READ: need to specify a time level for data with time axis')
271
272 if( .NOT.module_is_initialized )call mpp_error( FATAL, 'MPP_READ: must first call mpp_io_init.' )
273 if( .NOT.mpp_file(unit)%opened )call mpp_error( FATAL, 'MPP_READ: invalid unit number.' )
274
275 call mpp_get_compute_domain( domain, is, ie, js, je, tile_count=tile_count )
276 call mpp_get_data_domain ( domain, isd, ied, jsd, jed, x_is_global=x_is_global, &
277 y_is_global=y_is_global, tile_count=tile_count )
278 call mpp_get_memory_domain ( domain, ism, iem, jsm, jem )
279 call mpp_get_global_domain ( domain, isg, ieg, jsg, jeg, tile_count=tile_count )
280
281 ! when domain is symmetry, extra point is needed for some data on x/y direction
282 position = CENTER
283 if(mpp_domain_is_symmetry(domain)) then
284 if( size(data,1).EQ.ie-is+1 .AND. size(data,2).EQ.je-js+1 ) then ! CENTER
285 data_has_halos = .FALSE.
286 else if( size(data,1).EQ.ie-is+2 .AND. size(data,2).EQ.je-js+1 ) then ! EAST
287 data_has_halos = .FALSE.
288 position = EAST
289 ie = ie + 1
290 else if( size(data,1).EQ.ie-is+1 .AND. size(data,2).EQ.je-js+2 ) then ! NORTH
291 position = NORTH
292 data_has_halos = .FALSE.
293 je = je + 1
294 else if( size(data,1).EQ.ie-is+2 .AND. size(data,2).EQ.je-js+2 ) then ! CORNER
295 position = CORNER
296 data_has_halos = .FALSE.
297 ie = ie + 1; je = je + 1
298 else if( size(data,1).EQ.iem-ism+1 .AND. size(data,2).EQ.jem-jsm+1 )then ! CENTER
299 data_has_halos = .TRUE.
300 else if( size(data,1).EQ.iem-ism+2 .AND. size(data,2).EQ.jem-jsm+1 )then ! EAST
301 position = EAST
302 data_has_halos = .TRUE.
303 ie = ie + 1
304 else if( size(data,1).EQ.iem-ism+1 .AND. size(data,2).EQ.jem-jsm+2 )then ! NORTH
305 position = NORTH
306 data_has_halos = .TRUE.
307 je = je + 1
308 else if( size(data,1).EQ.iem-ism+2 .AND. size(data,2).EQ.jem-jsm+2 )then ! CORNER
309 position = CORNER
310 data_has_halos = .TRUE.
311 ie = ie + 1; je = je + 1
312 else
313 call mpp_error( FATAL, 'MPP_READ: when domain is symmetry, data must be either on ' &
314 //'compute domain or data domain with the consideration of shifting.' )
315 end if
316 else
317 if( size(data,1).EQ.ie-is+1 .AND. size(data,2).EQ.je-js+1 )then
318 data_has_halos = .FALSE.
319 else if( size(data,1).EQ.iem-ism+1 .AND. size(data,2).EQ.jem-jsm+1 )then
320 data_has_halos = .TRUE.
321 else
322 call mpp_error( FATAL, 'MPP_READ: data must be either on compute domain or data domain.' )
323 end if
324 endif
325 halos_are_global = x_is_global .AND. y_is_global
326 if( npes.GT.1 .AND. mpp_file(unit)%threading.EQ.MPP_SINGLE )then
327 if( halos_are_global )then !you can read directly into data array
328 if( pe.EQ.0 )call READ_RECORD_( unit, field, size(data(:,:,:)), data, tindex )
329 else
330 lenx=size(data,1)
331 leny=size(data,2)
332 lenz=size(data,3)
333 len=lenx*leny*lenz
334 allocate(gdata(len))
335! read field on pe 0 and pass to all pes
336 if( pe.EQ.0 ) call READ_RECORD_( unit, field, len, gdata, tindex )
337! broadcasting global array, this can be expensive!
338 call mpp_transmit( put_data=gdata(1), plen=len, to_pe=ALL_PES, &
339 get_data=gdata(1), glen=len, from_pe=0 )
340 ioff = is; joff = js
341 if( data_has_halos )then
342 ioff = isd; joff = jsd
343 end if
344 do k=1,size(data,3)
345 do j=js,je
346 do i=is,ie
347 n=(i-isg+1) + (j-jsg)*lenx + (k-1)*lenx*leny
348 data(i-ioff+1,j-joff+1,k)=gdata(n)
349 enddo
350 enddo
351 enddo
352 deallocate(gdata)
353 call mpp_sync_self() ! ensure MPI_ISEND is done.
354 end if
355 else if( data_has_halos )then
356! for uniprocessor or multithreaded read
357! read compute domain as contiguous data
358
359 allocate( cdata(is:ie,js:je,size(data,3)) )
360 call READ_RECORD_(unit,field,size(cdata(:,:,:)),cdata,tindex,domain,position,tile_count)
361
362 data(is-isd+1:ie-isd+1,js-jsd+1:je-jsd+1,:) = cdata(:,:,:)
363 deallocate(cdata)
364 else
365 call READ_RECORD_(unit,field,size(data(:,:,:)),data,tindex,domain,position,tile_count)
366 end if
367
368 call mpp_clock_end(mpp_read_clock)
369
370 return
371 end subroutine MPP_READ_2DDECOMP_3D_

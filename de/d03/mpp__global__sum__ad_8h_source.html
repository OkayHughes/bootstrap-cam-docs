23subroutine MPP_GLOBAL_SUM_AD_( domain, field, gsum_, flags, position, tile_count, overflow_check)
24 MPP_TYPE_ :: MPP_GLOBAL_SUM_
25 type(domain2D), intent(in) :: domain
26 MPP_TYPE_, intent(inout) :: field(:,: MPP_EXTRA_INDICES_ )
27 MPP_TYPE_, intent(in) :: gsum_
28 integer, intent(in), optional :: flags
29 integer, intent(in), optional :: position
30 integer, intent(in), optional :: tile_count
31 logical, intent(in), optional :: overflow_check
32
33 MPP_TYPE_, dimension(:,:), allocatable :: field2D
34 MPP_TYPE_, dimension(:,:), allocatable :: global2D
35 MPP_TYPE_, dimension(MAX_TILES), save :: gsum, nbrgsum, mygsum
36
37 integer :: i,j, ioff,joff, isc, iec, jsc, jec, is, ie, js, je, ishift, jshift, ioffset, joffset
38 integer :: gxsize, gysize
39 integer :: global_flag, tile, ntile, nlist, n, list, m
40
41 if( domain%max_ntile_pe > MAX_TILES ) call mpp_error(FATAL, "MPP_GLOBAL_SUM: number of tiles is exceed MAX_TILES")
42 ntile = size(domain%x(:))
43 nlist = size(domain%list(:))
44 tile = 1
45 if(present(tile_count)) tile = tile_count
46 global_flag = NON_BITWISE_EXACT_SUM
47 if(present(flags)) global_flag = flags
48
49 call mpp_get_domain_shift(domain, ishift, jshift, position)
50
51 if( size(field,1).EQ.domain%x(tile)%compute%size+ishift .AND. size(field,2).EQ.domain%y(tile)%compute%size+jshift )then
52!field is on compute domain
53 ioff = -domain%x(tile)%compute%begin + 1
54 joff = -domain%y(tile)%compute%begin + 1
55 else if( size(field,1).EQ.domain%x(tile)%memory%size+ishift .AND. size(field,2).EQ.domain%y(tile)%memory%size+jshift )then
56!field is on data domain
57 ioff = -domain%x(tile)%data%begin + 1
58 joff = -domain%y(tile)%data%begin + 1
59 else
60 call mpp_error( FATAL, 'MPP_GLOBAL_SUM_: incoming field array must match either compute domain or data domain.' )
61 end if
62
63 if(domain%ntiles > MAX_TILES) call mpp_error( FATAL, &
64 'MPP_GLOBAL_SUM_: number of tiles on this mosaic is greater than MAXTILES')
65
66 call mpp_get_compute_domain( domain, is, ie, js, je, tile_count = tile_count )
67 isc = is; iec = ie + ishift; jsc = js; jec = je + jshift
68
69 call mpp_get_global_domain(domain, xsize = gxsize, ysize = gysize )
70 MPP_GLOBAL_SUM_ = 0
71 if( global_flag == BITWISE_EXACT_SUM )then
72 !this is bitwise exact across different PE counts.
73
74 allocate( field2D (isc:iec,jsc:jec) )
75 allocate( global2D( gxsize+ishift, gysize+jshift ) )
76 field2D = 0.
77 ioffset = domain%x(tile)%goffset*ishift; joffset = domain%y(tile)%goffset*jshift
78 if( tile == ntile) then
79 if(domain%ntiles == 1 ) then
80 MPP_GLOBAL_SUM_ = gsum_
81 mygsum(tile) = MPP_GLOBAL_SUM_
82 else if( nlist == 1) then
83 MPP_GLOBAL_SUM_ = gsum_
84 mygsum(1:ntile) = MPP_GLOBAL_SUM_
85 else ! need to sum by the order of tile_count
86 MPP_GLOBAL_SUM_ = gsum_
87 gsum(1:domain%ntiles) = MPP_GLOBAL_SUM_
88 !--- receive data from root_pe of each tile
89 do list = 1, nlist - 1
90 m = mod( domain%pos+nlist-list, nlist )
91 if( domain%list(m)%pe == domain%list(m)%tile_root_pe ) then
92 do n = 1, size(domain%list(m)%x(:))
93 nbrgsum(n) = gsum(domain%list(m)%tile_id(n))
94 end do
95 call mpp_send( nbrgsum(1), plen=size(domain%list(m)%x(:)), to_pe=domain%list(m)%pe, tag=COMM_TAG_1)
96 end if
97 end do
98 !--- send the data to other pe if the current pe is the root pe of any tile
99 if( mpp_domain_is_tile_root_pe(domain) ) then
100 do list = 1, nlist - 1
101 m = mod( domain%pos+list, nlist )
102 call mpp_recv( mygsum(1), glen=ntile, from_pe=domain%list(m)%pe, tag=COMM_TAG_1 )
103 end do
104 end if
105 call mpp_sync_self()
106 ! first fill the global sum on current pe.
107 do n = 1, ntile
108 mygsum(n) = gsum(domain%tile_id(n))
109 end do
110
111 end if
112 end if
113 global2D(1:gxsize+ioffset,1:gysize+joffset) = mygsum(tile)
114 if ( present( tile_count ) ) then
115 call mpp_global_field_ad( domain, field2D, global2D, position=position, tile_count=tile_count )
116 else
117 call mpp_global_field_ad( domain, field2D, global2D, position=position )
118 endif
119
120 do j = jsc, jec
121 do i = isc, iec
122 field(i+ioff:i+ioff,j+joff:j+joff MPP_EXTRA_INDICES_) = field2D(i,j)
123 end do
124 end do
125
126 deallocate(global2D, field2d)
127 else if ( global_flag == BITWISE_EFP_SUM )then
128#ifdef DO_EFP_SUM_
129 !this is bitwise across different PE counts using EFP sum
130 if( ntile > 1 ) then
131 call mpp_error( FATAL, 'MPP_GLOBAL_SUM_: multiple tile per pe is not supported for BITWISE_EFP_SUM')
132 endif
133
134 do j = jsc, jec
135 do i = isc, iec
136 field(i+ioff:i+ioff,j+joff:j+joff MPP_EXTRA_INDICES_) = gsum_
137 end do
138 end do
139#else
140 call mpp_error( FATAL, 'MPP_GLOBAL_SUM_: BITWISE_EFP_SUM is only implemented for real number, contact developer')
141#endif
142 else !this is not bitwise-exact across different PE counts
143 ioffset = domain%x(tile)%loffset*ishift; joffset = domain%y(tile)%loffset*jshift
144 mygsum(1:ntile) = 0
145 if(tile == ntile) then
146 MPP_GLOBAL_SUM_ = gsum_
147 call mpp_sum_ad( MPP_GLOBAL_SUM_, domain%list(:)%pe )
148 mygsum(1:ntile) = MPP_GLOBAL_SUM_
149 end if
150 field(is+ioff:ie+ioff+ioffset, js+joff:je+joff+joffset MPP_EXTRA_INDICES_) = mygsum(tile)
151 end if
152
153 return
154 end subroutine MPP_GLOBAL_SUM_AD_

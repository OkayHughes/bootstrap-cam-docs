19 subroutine WRITE_RECORD_( unit, field, nwords, data, time_in, domain, tile_count)
20!routine that is finally called by all mpp_write routines to perform the write
21!a non-netCDF record contains:
22! field ID
23! a set of 4 coordinates (is:ie,js:je) giving the data subdomain
24! a timelevel and a timestamp (=NULLTIME if field is static)
25! 3D real data (stored as 1D)
26!if you are using direct access I/O, the RECL argument to OPEN must be large enough for the above
27!in a global direct access file, record position on PE is given by %record.
28
29!Treatment of timestamp:
30! We assume that static fields have been passed without a timestamp.
31! Here that is converted into a timestamp of NULLTIME.
32! For non-netCDF fields, field is treated no differently, but is written
33! with a timestamp of NULLTIME. There is no check in the code to prevent
34! the user from repeatedly writing a static field.
35
36 integer, intent(in) :: unit, nwords
37 type(fieldtype), intent(in) :: field
38 MPP_TYPE_, intent(in) :: data(nwords)
39 MPP_TYPE_, intent(in), optional :: time_in
40 type(domain2D), intent(in), optional :: domain
41 integer, intent(in), optional :: tile_count
42 integer, dimension(size(field%axes(:))) :: start, axsiz
43 real(DOUBLE_KIND) :: time
44 integer :: time_level
45 logical :: newtime
46 integer :: subdomain(4)
47 integer :: packed_data(nwords)
48 integer :: i, is, ie, js, je
49
50 real(FLOAT_KIND) :: data_r4(nwords)
51 pointer( ptr1, data_r4)
52 pointer( ptr2, packed_data)
53
54 if (mpp_io_stack_size < nwords) call mpp_io_set_stack_size(nwords)
55
56 if( .NOT.module_is_initialized )call mpp_error( FATAL, 'MPP_WRITE: must first call mpp_io_init.' )
57 if( .NOT.mpp_file(unit)%write_on_this_pe) return
58 if( .NOT.mpp_file(unit)%opened )call mpp_error( FATAL, 'MPP_WRITE: invalid unit number.' )
59 if( .NOT.mpp_file(unit)%initialized )then
60!this is the first call to mpp_write
61!we now declare the file to be initialized
62!if this is netCDF we switch file from DEFINE mode to DATA mode
63 if( mpp_file(unit)%format.EQ.MPP_NETCDF )then
64#ifdef use_netCDF
65!NOFILL is probably required for parallel: any circumstances in which not advisable?
66 error = NF_SET_FILL( mpp_file(unit)%ncid, NF_NOFILL, i ); call netcdf_err( error, mpp_file(unit) )
67 if( mpp_file(unit)%action.EQ.MPP_WRONLY )then
68 if(header_buffer_val>0) then
69 error = NF__ENDDEF(mpp_file(unit)%ncid,header_buffer_val,4,0,4)
70 else
71 error = NF_ENDDEF(mpp_file(unit)%ncid)
72 endif
73 endif
74 call netcdf_err( error, mpp_file(unit) )
75#endif
76 else
77 call mpp_write_meta( unit, 'END', cval='metadata' )
78 end if
79 mpp_file(unit)%initialized = .TRUE.
80 if( verbose )print '(a,i6,a)', 'MPP_WRITE: PE=', pe, ' initialized file '//trim(mpp_file(unit)%name)//'.'
81 end if
82
83!initialize time: by default assume NULLTIME
84 time = NULLTIME
85 time_level = -1
86 newtime = .FALSE.
87 if( PRESENT(time_in) )time = time_in
88!increment time level if new time
89 if( time.GT.mpp_file(unit)%time+EPSILON(time) )then !new time
90 mpp_file(unit)%time_level = mpp_file(unit)%time_level + 1
91 mpp_file(unit)%time = time
92 newtime = .TRUE.
93 end if
94 if( verbose )print '(a,2i6,2i5,es13.5)', 'MPP_WRITE: PE, unit, %id, %time_level, %time=',&
95 pe, unit, mpp_file(unit)%id, mpp_file(unit)%time_level, mpp_file(unit)%time
96
97 if( mpp_file(unit)%format.EQ.MPP_NETCDF )then
98 ptr2 = LOC(mpp_io_stack(1))
99!define netCDF data block to be written:
100! time axis: START = time level
101! AXSIZ = 1
102! space axis: if there is no domain info
103! START = 1
104! AXSIZ = field%size(axis)
105! if there IS domain info:
106! start of domain is compute%start_index for multi-file I/O
107! global%start_index for all other cases
108! this number must be converted to 1 for NF_PUT_VAR
109! (netCDF fortran calls are with reference to 1),
110! So, START = compute%start_index - <start of domain> + 1
111! AXSIZ = usually compute%size
112! However, if compute%start_index-compute%end_index+1.NE.compute%size,
113! we assume that the call is passing a subdomain.
114! To pass a subdomain, you must pass a domain2D object that satisfies the following:
115! global%start_index must contain the <start of domain> as defined above;
116! the data domain and compute domain must refer to the subdomain being passed.
117! In this case, START = compute%start_index - <start of domain> + 1
118! AXSIZ = compute%start_index - compute%end_index + 1
119! NOTE: passing of subdomains will fail for multi-PE single-threaded I/O,
120! since that attempts to gather all data on PE 0.
121 start = 1
122 do i = 1,size(field%axes(:))
123 axsiz(i) = field%size(i)
124 if( i.EQ.field%time_axis_index )start(i) = mpp_file(unit)%time_level
125 start(i) = max(start(i),1)
126 end do
127
128 if( debug )print '(a,2i6,12i6)', 'WRITE_RECORD: PE, unit, start, axsiz=', pe, unit, start, axsiz
129#ifdef use_netCDF
130!write time information if new time
131 if( newtime )then
132 if( KIND(time).EQ.DOUBLE_KIND )then
133 error = NF_PUT_VAR1_DOUBLE( mpp_file(unit)%ncid, mpp_file(unit)%id, mpp_file(unit)%time_level, time )
134 else if( KIND(time).EQ.FLOAT_KIND )then
135 error = NF_PUT_VAR1_REAL ( mpp_file(unit)%ncid, mpp_file(unit)%id, mpp_file(unit)%time_level, time )
136 end if
137 end if
138 if( field%pack == 0 )then
139 packed_data = CEILING(data)
140 error = NF_PUT_VARA_INT ( mpp_file(unit)%ncid, field%id, start, axsiz, packed_data )
141 elseif( field%pack.GT.0 .and. field%pack.LE.2 )then
142 if( KIND(data).EQ.DOUBLE_KIND )then
143 error = NF_PUT_VARA_DOUBLE( mpp_file(unit)%ncid, field%id, start, axsiz, data )
144 else if( KIND(data).EQ.FLOAT_KIND )then
145 error = NF_PUT_VARA_REAL ( mpp_file(unit)%ncid, field%id, start, axsiz, data )
146 end if
147 else !convert to integer using scale and add: no error check on packed data representation
148 packed_data = nint((data-field%add)/field%scale)
149 error = NF_PUT_VARA_INT ( mpp_file(unit)%ncid, field%id, start, axsiz, packed_data )
150 end if
151 call netcdf_err( error, mpp_file(unit), field=field )
152#endif
153 else !non-netCDF
154 ptr1 = LOC(mpp_io_stack(1))
155!subdomain contains (/is,ie,js,je/)
156 if( PRESENT(domain) )then
157 call mpp_get_compute_domain(domain, is, ie, js, je)
158 subdomain(:) = (/ is, ie, js, je /)
159 else
160 subdomain(:) = -1 ! -1 means use global value from axis metadata
161 end if
162 if( mpp_file(unit)%format.EQ.MPP_ASCII )then
163!implies sequential access
164 write( unit,* )field%id, subdomain, time_level, time, data
165 else !MPP_IEEE32 or MPP_NATIVE
166 if( mpp_file(unit)%access.EQ.MPP_SEQUENTIAL )then
167#ifdef __sgi
168 if( mpp_file(unit)%format.EQ.MPP_IEEE32 )then
169 data_r4 = data !IEEE conversion layer on SGI until assign -N ieee_32 is supported
170 write(unit)field%id, subdomain, time_level, time, data_r4
171 else
172 write(unit)field%id, subdomain, time_level, time, data
173 end if
174#else
175 write(unit)field%id, subdomain, time_level, time, data
176#endif
177 else !MPP_DIRECT
178#ifdef __sgi
179 if( mpp_file(unit)%format.EQ.MPP_IEEE32 )then
180 data_r4 = data !IEEE conversion layer on SGI until assign -N ieee_32 is supported
181 write( unit, rec=mpp_file(unit)%record )field%id, subdomain, time_level, time, data_r4
182 else
183 write( unit, rec=mpp_file(unit)%record )field%id, subdomain, time_level, time, data
184 end if
185#else
186 write( unit, rec=mpp_file(unit)%record )field%id, subdomain, time_level, time, data
187#endif
188 if( debug )print '(a,i6,a,i6)', 'MPP_WRITE: PE=', pe, ' wrote record ', mpp_file(unit)%record
189 end if
190 end if
191 end if
192
193!recompute current record for direct access I/O
194 if( mpp_file(unit)%access.EQ.MPP_DIRECT )then
195 if( mpp_file(unit)%fileset.EQ.MPP_SINGLE )then
196!assumes all PEs participate in I/O: modify later
197 mpp_file(unit)%record = mpp_file(unit)%record + records_per_pe*npes
198 else
199 mpp_file(unit)%record = mpp_file(unit)%record + records_per_pe
200 end if
201 end if
202
203 return
204 end subroutine WRITE_RECORD_

223 subroutine MPP_WRITE_2DDECOMP_3D_( unit, field, domain, data, tstamp, tile_count, default_data)
224!mpp_write writes <data> which has the domain decomposition <domain>
225 integer, intent(in) :: unit
226 type(fieldtype), intent(in) :: field
227 type(domain2D), intent(inout) :: domain
228 MPP_TYPE_, intent(inout) :: data(:,:,:)
229 MPP_TYPE_, intent(in), optional :: tstamp
230 integer, intent(in), optional :: tile_count
231 MPP_TYPE_, intent(in), optional :: default_data
232
233!cdata is used to store compute domain as contiguous data
234!gdata is used to globalize data for multi-PE single-threaded I/O
235 MPP_TYPE_, allocatable, dimension(:,:,:) :: cdata, gdata
236!NEW: data may be on compute OR data domain
237 logical :: data_has_halos, halos_are_global, x_is_global, y_is_global
238 integer :: is, ie, js, je, isd, ied, jsd, jed, isg, ieg, jsg, jeg, ism, iem, jsm, jem
239 integer :: position, errunit
240 type(domain2d), pointer :: io_domain=>NULL()
241
242 call mpp_clock_begin(mpp_write_clock)
243
244 errunit = stderr()
245 if( .NOT.module_is_initialized )call mpp_error( FATAL, 'MPP_WRITE: must first call mpp_io_init.' )
246 if( .NOT.mpp_file(unit)%valid )call mpp_error( FATAL, 'MPP_WRITE: invalid unit number.' )
247
248 position = field%position
249
250 call mpp_get_compute_domain( domain, is, ie, js, je, tile_count=tile_count, position=position )
251 call mpp_get_data_domain ( domain, isd, ied, jsd, jed, x_is_global=x_is_global, &
252 y_is_global=y_is_global, tile_count=tile_count, position=position )
253 call mpp_get_memory_domain ( domain, ism, iem, jsm, jem, position=position )
254
255 if( size(data,1).EQ.ie-is+1 .AND. size(data,2).EQ.je-js+1 )then
256 data_has_halos = .FALSE.
257 else if( size(data,1).EQ.iem-ism+1 .AND. size(data,2).EQ.jem-jsm+1 )then
258 data_has_halos = .TRUE.
259 else
260 write( errunit,'(a,10i5)' )'MPP_WRITE_2DDECOMP fails on field '//trim(field%name)// &
261 ': is,ie,js,je, ism,iem,jsm,jem, size(data,1), size(data,2)=', &
262 is,ie,js,je, ism,iem,jsm,jem, size(data,1), size(data,2)
263 call mpp_error( FATAL, 'MPP_WRITE: data must be either on compute domain or data domain.' )
264 end if
265 halos_are_global = x_is_global .AND. y_is_global
266 if( npes.GT.1 .AND. mpp_file(unit)%threading.EQ.MPP_SINGLE )then
267 if( halos_are_global )then
268 call mpp_update_domains( data, domain, position = position )
269!all non-0 PEs have passed their data to PE 0 and may now exit
270 if(mpp_file(unit)%write_on_this_pe ) then
271 call WRITE_RECORD_( unit, field, size(data(:,:,:)), data, tstamp)
272 endif
273 else
274!put field onto global domain
275 call mpp_get_global_domain ( domain, isg, ieg, jsg, jeg, tile_count=tile_count, position=position )
276 if(mpp_file(unit)%write_on_this_pe .OR. .NOT. global_field_on_root_pe) then
277 allocate( gdata(isg:ieg,jsg:jeg,size(data,3)) )
278 else
279 allocate( gdata(1,1,1))
280 endif
281 if(global_field_on_root_pe) then
282 call mpp_global_field( domain, data, gdata, position = position, &
283 flags=XUPDATE+YUPDATE+GLOBAL_ROOT_ONLY, &
284 default_data=default_data)
285 else
286 call mpp_global_field( domain, data, gdata, position = position, &
287 default_data=default_data)
288 endif
289!all non-0 PEs have passed their data to PE 0 and may now exit
290 if(mpp_file(unit)%write_on_this_pe ) then
291 call WRITE_RECORD_( unit, field, size(gdata(:,:,:)), gdata, tstamp)
292 endif
293 deallocate(gdata)
294 end if
295 else if(mpp_file(unit)%io_domain_exist ) then
296 if( halos_are_global )then
297 call mpp_update_domains( data, domain, position = position )
298 if(mpp_file(unit)%write_on_this_pe ) then
299 call WRITE_RECORD_( unit, field, size(data(:,:,:)), data, tstamp)
300 endif
301 else
302 io_domain=>mpp_get_io_domain(mpp_file(unit)%domain)
303 call mpp_get_global_domain ( io_domain, isg, ieg, jsg, jeg, tile_count=tile_count, position=position )
304 if(mpp_file(unit)%write_on_this_pe .OR. .NOT. global_field_on_root_pe) then
305 allocate( gdata(isg:ieg,jsg:jeg,size(data,3)) )
306 else
307 allocate( gdata(1,1,1))
308 endif
309 if(global_field_on_root_pe) then
310 call mpp_global_field( io_domain, data, gdata, position = position, &
311 flags=XUPDATE+YUPDATE+GLOBAL_ROOT_ONLY, &
312 default_data=default_data)
313 else
314 call mpp_global_field( io_domain, data, gdata, position = position, &
315 default_data=default_data)
316 endif
317 io_domain => NULL()
318 if(mpp_file(unit)%write_on_this_pe ) then
319 call WRITE_RECORD_( unit, field, size(gdata(:,:,:)), gdata, tstamp)
320 endif
321 deallocate( gdata )
322 endif
323 else if( data_has_halos )then
324!store compute domain as contiguous data and pass to write_record
325 allocate( cdata(is:ie,js:je,size(data,3)) )
326 cdata(:,:,:) = data(is-isd+1:ie-isd+1,js-jsd+1:je-jsd+1,:)
327 call WRITE_RECORD_( unit, field, size(cdata(:,:,:)), cdata, tstamp, domain, tile_count )
328 else
329!data is already contiguous
330 call WRITE_RECORD_( unit, field, size(data(:,:,:)), data, tstamp, domain, tile_count )
331 end if
332
333 call mpp_clock_end(mpp_write_clock)
334
335 return
336 end subroutine MPP_WRITE_2DDECOMP_3D_

36 subroutine MPP_GLOBAL_FIELD_UG_3D_( domain, local, global, flags, default_data)
37!get a global field from a local field
38!local field may be on compute OR data domain
39 type(domainUG), intent(in) :: domain
40 MPP_TYPE_, intent(in) :: local(domain%compute%begin:,:)
41 MPP_TYPE_, intent(out) :: global(domain%global%begin:,:)
42 integer, intent(in), optional :: flags
43 MPP_TYPE_, intent(in), optional :: default_data
44
45 integer :: l, k, m, n, nd, nwords, lpos, rpos, ioff, joff, from_pe, tile_id
46 integer :: ke, lsc, lec, ls, le, nword_me
47 integer :: ipos, jpos
48 logical :: xonly, yonly, root_only, global_on_this_pe
49 MPP_TYPE_ :: clocal (domain%compute%size*size(local,2))
50 MPP_TYPE_ :: cremote(domain%compute%max_size*size(local,2))
51 integer :: stackuse
52 character(len=8) :: text
53
54 pointer( ptr_local, clocal )
55 pointer( ptr_remote, cremote )
56
57 stackuse = size(clocal(:))+size(cremote(:))
58 if( stackuse.GT.mpp_domains_stack_size )then
59 write( text, '(i8)' )stackuse
60 call mpp_error( FATAL, &
61 'MPP_DO_GLOBAL_FIELD_UG user stack overflow: call mpp_domains_set_stack_size('//trim(text)//') from all PEs.' )
62 end if
63 mpp_domains_stack_hwm = max( mpp_domains_stack_hwm, stackuse )
64
65 ptr_local = LOC(mpp_domains_stack)
66 ptr_remote = LOC(mpp_domains_stack(size(clocal(:))+1))
67
68 if( .NOT.module_is_initialized )call mpp_error( FATAL, 'MPP_GLOBAL_FIELD_UG: must first call mpp_domains_init.' )
69
70 root_only = .FALSE.
71 if( PRESENT(flags) ) root_only = BTEST(flags, ROOT_GLOBAL)
72
73 global_on_this_pe = .NOT. root_only .OR. domain%pe == domain%tile_root_pe
74 ipos = 0; jpos = 0
75 if(global_on_this_pe ) then
76 if(size(local,2).NE.size(global,2) ) call mpp_error( FATAL, &
77 'MPP_GLOBAL_FIELD_UG: mismatch of third dimension size of global and local')
78 if( size(global,1).NE.domain%global%size)then
79 call mpp_error( FATAL, 'MPP_GLOBAL_FIELD_UG: incoming arrays do not match domain.' )
80 endif
81 endif
82
83 if( size(local,1).NE.domain%compute%size )then
84 call mpp_error( FATAL, 'MPP_GLOBAL_FIELD_UG_: incoming field array must match compute domain ' )
85 end if
86
87 ke = size(local,2)
88 lsc = domain%compute%begin; lec = domain%compute%end
89
90 nword_me = (lec-lsc+1)*ke
91
92! make contiguous array from compute domain
93 m = 0
94 if(global_on_this_pe) then
95 if(PRESENT(default_data)) then
96 global = default_data
97 else
98 global = 0
99 endif
100
101 do k = 1, ke
102 do l = lsc, lec
103 m = m + 1
104 clocal(m) = local(l,k)
105 global(l,k) = clocal(m) !always fill local domain directly
106 end do
107 end do
108 else
109 do k = 1, ke
110 do l = lsc, lec
111 m = m + 1
112 clocal(m) = local(l,k)
113 end do
114 end do
115 endif
116
117 !fill off-domains (note loops begin at an offset of 1)
118 tile_id = domain%tile_id
119 nd = size(domain%list(:))
120 if(root_only) then
121 if(domain%pe .NE. domain%tile_root_pe) then
122 call mpp_send( clocal(1), plen=nword_me, to_pe=domain%tile_root_pe, tag=COMM_TAG_1 )
123 else
124 do n = 1,nd-1
125 rpos = mod(domain%pos+n,nd)
126 if( domain%list(rpos)%tile_id .NE. tile_id ) cycle
127 nwords = domain%list(rpos)%compute%size * ke
128 call mpp_recv(cremote(1), glen=nwords, from_pe=domain%list(rpos)%pe, tag=COMM_TAG_1 )
129 m = 0
130 ls = domain%list(rpos)%compute%begin; le = domain%list(rpos)%compute%end
131
132 do k = 1,ke
133 do l = ls, le
134 m = m + 1
135 global(l,k) = cremote(m)
136 end do
137 end do
138 end do
139 endif
140 else
141 do n = 1,nd-1
142 lpos = mod(domain%pos+nd-n,nd)
143 if( domain%list(lpos)%tile_id.NE. tile_id ) cycle ! global field only within tile
144 call mpp_send( clocal(1), plen=nword_me, to_pe=domain%list(lpos)%pe, tag=COMM_TAG_2 )
145 end do
146 do n = 1,nd-1
147 rpos = mod(domain%pos+n,nd)
148 if( domain%list(rpos)%tile_id .NE. tile_id ) cycle ! global field only within tile
149 nwords = domain%list(rpos)%compute%size * ke
150 call mpp_recv( cremote(1), glen=nwords, from_pe=domain%list(rpos)%pe, tag=COMM_TAG_2 )
151 m = 0
152 ls = domain%list(rpos)%compute%begin; le = domain%list(rpos)%compute%end
153
154 do k = 1,ke
155 do l = ls, le
156 m = m + 1
157 global(l,k) = cremote(m)
158 end do
159 end do
160 enddo
161 endif
162
163 call mpp_sync_self()
164
165 end subroutine MPP_GLOBAL_FIELD_UG_3D_

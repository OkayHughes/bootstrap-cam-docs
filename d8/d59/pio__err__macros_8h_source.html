20/* Err is used to keep track of errors within each set of tests,
21 * total_err is the number of errors in the entire test program, which
22 * generally cosists of several sets of tests. */
23static int total_err = 0, err = 0;
24
25/* This macro prints an error message with line number and name of
26 * test program. */
27#define PERR do { \
28 fflush(stdout); /* Make sure our stdout is synced with stderr. */ \
29 err++; \
30 fprintf(stderr, "Sorry! Unexpected result, %s, line: %d\n", \
31 __FILE__, __LINE__); \
32 fflush(stderr); \
33 return 2; \
34 } while (0)
35
36/* After a set of tests, report the number of errors, and increment
37 * total_err. */
38#define PSUMMARIZE_ERR do { \
39 if (err) \
40 { \
41 printf("%d failures\n", err); \
42 total_err += err; \
43 err = 0; \
44 } \
45 else \
46 if (!my_rank) \
47 printf("ok.\n"); \
48 } while (0)
49
50/* This macro prints out our total number of errors, if any, and exits
51 * with a 0 if there are not, or a 2 if there were errors. Make will
52 * stop if a non-zero value is returned from a test program. */
53#define PFINAL_RESULTS do { \
54 if (total_err) \
55 { \
56 printf("%d errors detected! Sorry!\n", total_err); \
57 return 2; \
58 } \
59 if (!my_rank) \
60 printf("*** Tests successful!\n\n"); \
61 return 0; \
62 } while (0)

66#endif /* _PIO_ERR_MACROS_H */

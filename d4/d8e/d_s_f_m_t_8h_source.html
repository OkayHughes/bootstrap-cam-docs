239inline static uint32_t dsfmt_genrand_uint32(dsfmt_t *dsfmt) {
240 uint32_t r;

244 dsfmt_gen_rand_all(dsfmt);

348inline static double dsfmt_genrand_open_open(dsfmt_t *dsfmt) {
349 double *dsfmt64 = &dsfmt->status[0].d[0];

351 double d;

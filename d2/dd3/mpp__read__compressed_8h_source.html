34 subroutine MPP_READ_COMPRESSED_2D_(unit, field, domain, data, tindex, start, nread, threading)

49 call mpp_clock_begin(mpp_read_clock)

53
54 print_compressed_chksum = .FALSE.
55
56 if(size(data) > 0) then
57 data = 0 !! zero out data so other tiles do not contribute junk to chksum
58 threading_flag = MPP_SINGLE
59 if( PRESENT(threading) )threading_flag = threading
60 if( threading_flag == MPP_MULTI ) then
61 call read_record(unit,field,size(data(:,:)),data,tindex,start_in=start, axsiz_in=nread)
62 else if( threading_flag == MPP_SINGLE ) then
63
64 io_domain=>mpp_get_io_domain(domain)
65 if(.not. ASSOCIATED(io_domain)) call mpp_error( FATAL, 'MPP_READ_COMPRESSED_2D_: io_domain must be defined.' )
66 npes = mpp_get_domain_npes(io_domain)
67 allocate(pelist(npes))
68 call mpp_get_pelist(io_domain,pelist)
69
70 if(mpp_pe() == pelist(1)) call read_record(unit,field,size(data(:,:)),data,tindex,start_in=start, axsiz_in=nread)
71
72 !--- z1l replace mpp_broadcast with mpp_send/mpp_recv to avoid hang in calling MPI_COMM_CREATE
73 !--- because size(pelist) might be different for different rank.
74 !--- prepost receive
75 if( mpp_pe() == pelist(1) ) then
76 do p = 2, npes
77 call mpp_send(data(1,1), plen=size(data(:,:)), to_pe=pelist(p), tag=COMM_TAG_1)
78 enddo
79 call mpp_sync_self()
80 else
81 call mpp_recv(data(1,1), glen=size(data(:,:)), from_pe=pelist(1), block=.false., tag=COMM_TAG_1)
82 call mpp_sync_self(check=EVENT_RECV)
83 endif
84
85 deallocate(pelist)
86 else
87 call mpp_error( FATAL, 'MPP_READ_COMPRESSED_2D_: threading should be MPP_SINGLE or MPP_MULTI')
88 endif
89 endif
90
91 compute_chksum = .FALSE.
92 if (ANY(field%checksum /= default_field%checksum) ) compute_chksum = .TRUE.
93
94 if (compute_chksum) then
95#ifdef use_netCDF
96 if (field%type==NF_INT) then
97 if (field%fill == MPP_FILL_DOUBLE .or. field%fill == real(MPP_FILL_INT) ) then
98 chk = mpp_chksum( ceiling(data), mask_val=MPP_FILL_INT )
99 else
100 call mpp_error(NOTE,"During mpp_io(mpp_read_compressed_2d) int field "//trim(field%name)// &
101 " found fill. Icebergs, or code using defaults can safely ignore. "// &
102 " If manually overriding compressed restart fills, confirm this is what you want.")
103 chk = mpp_chksum( ceiling(data), mask_val=field%fill)
104 end if
105 else !!real data
106 chk = mpp_chksum(data,mask_val=field%fill)
107 end if
108#endif
109 !!compare
110 if ( print_compressed_chksum) then
111 if ( mpp_pe() == mpp_root_pe() ) then
112 print '(A,Z16)', "mpp_read_compressed_2d chksum: "//trim(field%name)//" = ", chk
113 !! discuss making fatal after testing/review to match other routines.
114 !! Need to do some nword-counting and digging with pjp
115 !! this should be if ( chk /= field%checksum ) as it was tested @ulm_201505..
116 if ( MOD(chk, field%checksum(1)) /= 0 ) then
117 print '(A,Z16)', "File stored checksum: "//trim(field%name)//" = ", field%checksum(1)
118 call mpp_error(NOTE,"mpp_read_compressed_2d chksum: "//trim(field%name)//" failed!")
119 end if
120 endif
121 end if
122 end if
123
124 call mpp_clock_end(mpp_read_clock)
125 return
126 end subroutine MPP_READ_COMPRESSED_2D_

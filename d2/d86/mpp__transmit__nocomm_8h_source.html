25 subroutine MPP_TRANSMIT_( put_data, put_len, to_pe, get_data, get_len, from_pe, block, tag, recv_request, send_request )
26!a message-passing routine intended to be reminiscent equally of both MPI and SHMEM
27
28!put_data and get_data are contiguous MPP_TYPE_ arrays
29
30!at each call, your put_data array is put to to_pe's get_data
31! your get_data array is got from from_pe's put_data
32!i.e we assume that typically (e.g updating halo regions) each PE performs a put _and_ a get
33
34!special PE designations:
35! NULL_PE: to disable a put or a get (e.g at boundaries)
36! ANY_PE: if remote PE for the put or get is to be unspecific
37! ALL_PES: broadcast and collect operations (collect not yet implemented)
38
39!ideally we would not pass length, but this f77-style call performs better (arrays passed by address, not descriptor)
40!further, this permits <length> contiguous words from an array of any rank to be passed (avoiding f90 rank conformance check)
41
42!caller is responsible for completion checks (mpp_sync_self) before and after
43
44 integer, intent(in) :: put_len, to_pe, get_len, from_pe
45 MPP_TYPE_, intent(in) :: put_data(*)
46 MPP_TYPE_, intent(out) :: get_data(*)
47 logical, intent(in), optional :: block
48 integer, intent(in), optional :: tag
49 integer, intent(out), optional :: recv_request, send_request
50
51 integer :: i, outunit
52 MPP_TYPE_, allocatable, save :: local_data(:) !local copy used by non-parallel code (no SHMEM or MPI)
53 integer(LONG_KIND), save :: get_data_addr=-9999
54 MPP_TYPE_ :: get_data_local(get_len_nocomm)
55 pointer(ptr_get_data, get_data_local)
56
57
58 if( .NOT.module_is_initialized )call mpp_error( FATAL, 'MPP_TRANSMIT: You must first call mpp_init.' )
59 if( to_pe.EQ.NULL_PE .AND. from_pe.EQ.NULL_PE )return
60
61 outunit = stdout()
62 if( debug )then
63 call SYSTEM_CLOCK(tick)
64 write( outunit,'(a,i18,a,i5,a,2i5,2i8)' )&
65 'T=',tick, ' PE=',pe, ' MPP_TRANSMIT begin: to_pe, from_pe, put_len, get_len=', to_pe, from_pe, put_len, get_len
66 end if
67
68!do put first and then get
69 if( to_pe.GE.0 .AND. to_pe.LT.npes )then
70 if( allocated(local_data) ) &
71 call mpp_error( FATAL, 'MPP_TRANSMIT: local_data should have been deallocated by prior receive.' )
72 if( get_len_nocomm > 0) then ! pre-post recv
73 ptr_get_data = get_data_addr
74 do i = 1,get_len_nocomm
75 get_data_local(i) = put_data(i)
76 end do
77 get_len_nocomm = 0
78 get_data_addr = -9999
79 else
80 allocate( local_data(put_len) )
81 do i = 1,put_len
82 local_data(i) = put_data(i)
83 end do
84 endif
85 else if( to_pe.EQ.ALL_PES )then !this is a broadcast from from_pe
86 if( from_pe.LT.0 .OR. from_pe.GE.npes )call mpp_error( FATAL, 'MPP_TRANSMIT: broadcasting from invalid PE.' )
87 if( put_len.GT.get_len )call mpp_error( FATAL, 'MPP_TRANSMIT: size mismatch between put_data and get_data.' )
88 if( pe.EQ.from_pe )then
89 if( LOC(get_data).NE.LOC(put_data) )then
90!dir$ IVDEP
91 do i = 1,get_len
92 get_data(i) = put_data(i)
93 end do
94 end if
95 end if
96 call mpp_broadcast( get_data, get_len, from_pe )
97 return
98
99 else if( to_pe.EQ.ANY_PE )then !we don't have a destination to do puts to, so only do gets
100
101 else if( to_pe.NE.NULL_PE )then !no other valid cases except NULL_PE
102 call mpp_error( FATAL, 'MPP_TRANSMIT: invalid to_pe.' )
103 end if
104
105!do the get: for libSMA, a get means do a wait to ensure put on remote PE is complete
106 if( from_pe.GE.0 .AND. from_pe.LT.npes )then
107 if( .NOT.allocated(local_data) ) then
108 get_data_addr = LOC(get_data)
109 get_len_nocomm = get_len
110 else
111 do i = 1,get_len
112 get_data(i) = local_data(i)
113 end do
114 deallocate(local_data)
115 endif
116 else if( from_pe.EQ.ANY_PE )then
117
118 else if( from_pe.EQ.ALL_PES )then
119 call mpp_error( FATAL, 'MPP_TRANSMIT: from_pe=ALL_PES has ambiguous meaning, and hence is not implemented.' )
120
121 else if( from_pe.NE.NULL_PE )then !only remaining valid choice is NULL_PE
122 call mpp_error( FATAL, 'MPP_TRANSMIT: invalid from_pe.' )
123 end if
124
125 if( debug )then
126 call SYSTEM_CLOCK(tick)
127 write( outunit,'(a,i18,a,i5,a,2i5,2i8)' )&
128 'T=',tick, ' PE=',pe, ' MPP_TRANSMIT end: to_pe, from_pe, put_len, get_len=', to_pe, from_pe, put_len, get_len
129 end if
130 return
131 end subroutine MPP_TRANSMIT_
